🔐 2. TENANT CONTEXT (القلب الحقيقي للنظام)

يتم فرضه على كل Request

export function getTenantContext(req) {
  const tenantId = req.user?.tenant_id;
  if (!tenantId) {
    throw new Error("Missing tenant context");
  }
  return { tenantId };
}

🚨 3. FORCED SCOPED DATABASE (أهم جزء)

بدل ما المطور ينسى tenant_id:

نخلي DB نفسها “تفرض العزل”

export function createScopedDB(tenantId: string) {
  return {
    cases: {
      findMany: (args = {}) =>
        db.cases.findMany({
          ...args,
          where: {
            ...args.where,
            tenant_id: tenantId
          }
        }),
      findFirst: (args = {}) =>
        db.cases.findFirst({
          ...args,
          where: {
            ...args.where,
            tenant_id: tenantId
          }
        })
    },
    clients: {
      findMany: (args = {}) =>
        db.clients.findMany({
          ...args,
          where: {
            ...args.where,
            tenant_id: tenantId
          }
        })
    }
  };
}

🧠 5. SERVICE LAYER ENFORCEMENT

كل Service يجب أن يكون Tenant-aware

export class CaseService {
  constructor(private db, private tenantId) {}
  async getCases() {
    return this.db.cases.findMany({
      where: { tenant_id: this.tenantId }
    });
  }
}

🔍 7. RUNTIME LEAK DETECTOR

يراقب أي خروج بيانات غير صحيح

function detectLeak(result, tenantId) {
  if (result?.some(r => r.tenant_id !== tenantId)) {
    throw new Error("DATA LEAK DETECTED");
  }
}

test("no cross tenant leakage", async () => {
  const t1Cases = await api.asTenant("A").getCases();
