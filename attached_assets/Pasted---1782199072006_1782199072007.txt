Case Analyzer

export async function analyzeCases(db) {
  const active =
    await db.case.count({
      where: {
        status: "ACTIVE"
      }
    });
  const overdue =
    await db.case.count({
      where: {
        deadline: {
          lt: new Date()
        },
        status: {
          not: "CLOSED"
        }
      }
    });
  return {
    active,
    overdue
  };
}

Lawyer Workload Scoring

export function calculateLawyerLoad({
  openCases,
  openTasks,
  hearings
}) {
  const score =
      openCases * 0.5
    + openTasks * 0.3
    + hearings * 0.2;
  return Math.min(
    Math.round(score),
    100
  );
}

⸻

Legal Risk Scoring

export function calculateLegalRisk({
  overdueCases,
  overdueTasks,
  pendingContracts
}) {
  let score = 0;
  score += overdueCases * 5;
  score += overdueTasks * 2;
  score += pendingContracts;
  return Math.min(score, 100);
}
