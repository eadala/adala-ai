⚙️ 7) Compiler (تحويل Graph إلى React)

export function compile(nodes: any) {
  return nodes.map((n: any) => {
    switch (n.type) {
      case "table":
        return `<CasesTable />`;
      case "stats":
        return `<StatsCard />`;
      case "card":
        return `<InfoCard />`;
    }
  }).join("\n");
}

⸻

🧠 8) AI داخل البناء (أهم ميزة)

النظام لا يبني UI فقط — بل يفهم القانون:

export function enhanceLegalUI(node: any) {
  if (node.type === "table" && node.props.source === "cases") {
    node.props.filters = ["status", "date", "client"];
  }
  if (node.type === "card") {
    node.props.aiInsights = true;
  }
  return node;
}
