# 1. جدول مزودي الاستضافة

## `shared/schema/hostingProviders.ts`

```ts
import {
  pgTable,
  uuid,
  varchar,
  boolean,
  text,
  timestamp
} from "drizzle-orm/pg-core";

export const hostingProviders = pgTable(
  "hosting_providers",
  {

    id: uuid("id")
      .defaultRandom()
      .primaryKey(),

    providerName: varchar(
      "provider_name",
      {
        length: 100
      }
    ).notNull(),

    apiKey: text(
      "api_key"
    ),

    secretKey: text(
      "secret_key"
    ),

    endpoint: text(
      "endpoint"
    ),

    isActive: boolean(
      "is_active"
    ).default(true),

    createdAt: timestamp(
      "created_at"
    ).defaultNow()

  }
);
```

# 4. Cloudflare Service

## `server/services/cloudflareService.ts`

```ts
import axios from "axios";

const baseURL =
"https://api.cloudflare.com/client/v4";

export async function createDNSRecord(

zoneId: string,

name: string,

ip: string

) {

  return axios.post(

    `${baseURL}/zones/${zoneId}/dns_records`,

    {

      type: "A",

      name,

      content: ip,

      proxied: true

    },

    {

      headers: {

        Authorization:

          `Bearer ${process.env.CLOUDFLARE_TOKEN}`,

        "Content-Type":

          "application/json"

      }

    }

  );

}
```

# 6. Hosting Service

## `server/services/hostingService.ts`

import os from "os";

export function getServerStatus() {

  return {

    cpu:

      os.cpus().length,

    memory:

      os.totalmem(),

    hostname:

      os.hostname(),

    platform:

      os.platform()

  };

}
