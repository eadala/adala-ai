🧠 2) AI Intent Parser (فهم الطلب)

📁 server/ai/parser.ts

export async function parseUIIntent(prompt: string) {
  return {
    type: "dashboard" | "form" | "table" | "landing",
    theme: "light",
    layout: "grid",
    components: extractComponents(prompt),
  };
}
function extractComponents(prompt: string) {
  const components = [];
  if (prompt.includes("قضايا")) components.push("cases_table");
  if (prompt.includes("عملاء")) components.push("clients_list");
  if (prompt.includes("إحصائيات")) components.push("stats_cards");
  return components;
}

🧩 6) Renderer (تحويل Schema إلى UI)

📁 client/ui/renderer.tsx

import { registry } from "./registry";
export function UIRenderer({ schema }: any) {
  return (
    <div className="p-6 space-y-4 bg-white min-h-screen">
      {schema.components.map((comp: any, i: number) => {
        const Component = registry[comp.type];
        if (!Component) return null;
        return <Component key={i} {...comp} />;
      })}
    </div>
  );
}

🧠 7) AI Builder API

📁 server/api/ui-builder.ts

import { generateUISchema } from "../ai/generator";
export async function uiBuilderRoute(req, res) {
  const { prompt } = req.body;
  const schema = await generateUISchema(prompt);
  res.json({
    success: true,
    schema,
  });
}

🧪 8) Preview Sandbox (Live Builder)

📁 client/pages/UIBuilder.tsx

import { useState } from "react";
import { UIRenderer } from "../ui/renderer";
export default function UIBuilder() {
  const [schema, setSchema] = useState(null);
  async function generate() {
    const res = await fetch("/api/ui-builder", {
      method: "POST",
      body: JSON.stringify({
        prompt: "صفحة إدارة القضايا مع إحصائيات وجدول",
      }),
    });
    const data = await res.json();
    setSchema(data.schema);
  }
  return (
    <div className="grid grid-cols-2 gap-4 p-4">
      <div className="border p-4">
        <button onClick={generate}>
          توليد واجهة
        </button>
      </div>
      <div className="border p-4">
        {schema && <UIRenderer schema={schema} />}
      </div>
    </div>
  );
}

🧠 9) Advanced AI Layer (اختياري لكن مهم)

تحسين الفهم القانوني:

export async function enhanceLegalUI(schema: any) {
  return {
    ...schema,
    components: schema.components.map((c: any) => {
      if (c.type === "table") {
        c.legalFilters = true;
      }
      if (c.type === "card") {
        c.aiInsights = true;
      }
      return c;
    }),
  };
}
