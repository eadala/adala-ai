export default function AlertPanel() {
  const alerts = [
    {
      type: "critical",
      message: "مكتب الرياض: ضغط موظفين عالي جدًا",
    },
    {
      type: "warning",
      message: "3 قضايا متأخرة تحتاج تدخل",
    },
    {
      type: "info",
      message: "تحسن أداء مكتب جدة بنسبة 18%",
    },
  ];
  return (
    <div className="bg-white p-4 rounded-xl shadow">
      <h2 className="font-bold mb-3">🚨 التنبيهات الذكية</h2>
      <div className="space-y-2">
        {alerts.map((a, i) => (
          <div
            key={i}
            className="p-2 rounded bg-gray-50 border"
          >
            {a.message}
          </div>
        ))}
      </div>
    </div>
  );
}
