🔥 1) إنشاء Tenant Context (الأساس)

import { createContext, useContext } from "react";
export type TenantContextType = {
  officeId: string;
  userId: string;
};
export const TenantContext = createContext<TenantContextType | null>(null);
export function useTenant() {
  const ctx = useContext(TenantContext);
  if (!ctx) throw new Error("TenantContext missing");
  return ctx;
}

🔥 2) Server-side Tenant Injector (القلب الحقيقي)

export function createTenantDB(db, officeId: string) {
  return {
    cases: {
      findMany: (query = {}) =>
        db.select()
          .from(cases)
          .where(and(
            eq(cases.officeId, officeId),
            query.where
          )),
      findOne: (id: string) =>
        db.select()
          .from(cases)
          .where(and(
            eq(cases.id, id),
            eq(cases.officeId, officeId)
          )),
      create: (data) =>
        db.insert(cases).values({
          ...data,
          officeId, // 🔥 forced injection
        }),
    },
    clients: {
      findMany: () =>
        db.select()
          .from(clients)
          .where(eq(clients.officeId, officeId)),
      create: (data) =>
        db.insert(clients).values({
          ...data,
          officeId,
        }),
    },
    documents: {
      findByCase: (caseId: string) =>
        db.select()
          .from(documents)
          .where(and(
            eq(documents.caseId, caseId),
            eq(documents.officeId, officeId)
          )),
    }
  };
}

🔒 4) Hard Guard (منع أي استعلام بدون officeId)

هذا أهم جزء:

export function assertTenantFilter(query: any, officeId: string) {
  const sql = query?.toString?.() || "";
  if (!sql.includes("office_id")) {
    throw new Error("SECURITY VIOLATION: Missing tenant filter");
  }
  return query;
}

🧱 5) Prisma/Drizzle-style enforcement (اختياري قوي)

إذا تريد مستوى أعلى:

export function strictDb(db, officeId: string) {
  return new Proxy(db, {
    get(target, prop) {
      const original = target[prop];
      if (typeof original === "function") {
        return (...args) => {
          const result = original.apply(target, args);
          // intercept SELECT queries
          return assertTenantFilter(result, officeId);
        };
      }
      return original;
    }
  });
}
