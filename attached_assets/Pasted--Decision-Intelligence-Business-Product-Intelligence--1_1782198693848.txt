📊 2. OBSERVABILITY COLLECTOR

import axios from "axios";
export async function collectSystemContext() {
  const query = async (q: string) => {
    const res = await axios.get("http://prometheus:9090/api/v1/query", {
      params: { query: q },
    });
    return parseFloat(res.data.data.result[0]?.value[1] || "0");
  };
  return {
    cpu: await query("node_cpu_usage_percent"),
    memory: await query("node_memory_usage_percent"),
    latency: await query("http_request_duration_ms_p95"),
    errors: await query("rate(http_requests_total{status=~'5..'}[5m])"),
    aiRequests: await query("ai_requests_total"),
    activeUsers: await query("active_users_total"),
  };
}

🧠 3. DECISION LAYER (Reactive Engine)

let cooldown = false;
export async function runDecisionLayer(ctx: any, prediction: any) {
  const risk =
    ctx.cpu * 0.25 +
    ctx.memory * 0.25 +
    ctx.latency * 0.3 +
    ctx.errors * 0.2;
  if (cooldown) return { status: "skipped" };
  if (risk > 80 || prediction.status === "CRITICAL") {
    cooldown = true;
    await fetch("http://api:8080/internal/heal", {
      method: "POST",
      headers: {
        Authorization: `Bearer ${process.env.HEAL_SECRET}`,
      },
    });
    setTimeout(() => (cooldown = false), 300000); // 5 min
  }
  return { risk };
}

🔮 4. INTELLIGENCE LAYER (Predictive Engine)

export async function runIntelligenceLayer(ctx: any) {
  const score =
    ctx.cpu * 0.3 +
    ctx.memory * 0.3 +
    ctx.latency * 0.2 +
    ctx.errors * 0.2;
  const status =
    score > 75 ? "CRITICAL" :
    score > 45 ? "WARNING" :
    "STABLE";
  return { score, status };
}

💰 5. BUSINESS LAYER (Cost + Workload)

export async function runBusinessLayer(ctx: any) {
  const workload =
    ctx.aiRequests * 0.4 +
    ctx.activeUsers * 0.3 +
    ctx.latency * 0.3;
  const cost = ctx.aiRequests * 0.01 + ctx.cpu * 0.02;
  return {
    workload:
      workload > 80 ? "HIGH" :
      workload > 40 ? "MEDIUM" :
      "LOW",
    cost,
  };
}

👤 6. PRODUCT LAYER (Revenue + UX + Legal Value)

export async function runProductLayer(ctx: any) {
  const intent =
    ctx.aiRequests > 100 ? "HIGH_INTENT" :
    ctx.activeUsers > 200 ? "EXPLORATION" :
    "CASUAL";
  const revenue =
    intent === "HIGH_INTENT" ? 10 :
    intent === "EXPLORATION" ? 3 :
    1;
  const legalValue = {
    efficiencyBoost: 0.18,
    winRateImpact: 0.12,
  };
  return {
    intent,
    revenue,
    legalValue,
  };
}

⚙️ 7. ACTION ENGINE (Unified Brain Executor)

async function executeActions(data: any) {
  const { prediction, decision, business, product } = data;
  // Cost optimization
  if (business.cost > 10) {
    console.log("Optimize resources triggered");
  }
  // UX adaptation
  if (product.intent === "HIGH_INTENT") {
    await fetch("http://api:8080/internal/personalize", {
      method: "POST",
      body: JSON.stringify({ mode: "premium_ai" }),
    });
  }
  // safety fallback
  if (prediction.status === "CRITICAL") {
    console.log("System in critical state");
  }
}
