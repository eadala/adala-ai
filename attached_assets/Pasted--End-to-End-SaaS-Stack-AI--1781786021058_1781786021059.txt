🧠 2) AI Gateway (قلب النظام)

🔥 ai-gateway package

export class AIGateway {
  constructor(
    private router: AIRouter,
    private policy: PolicyEngine,
    private billing: BillingService
  ) {}
  async execute(request: AIRequest) {
    const provider = this.policy.select(request);
    const result = await this.router.run(provider, request.prompt);
    await this.billing.log({
      tenantId: request.tenantId,
      provider,
      tokens: result.tokens,
      cost: this.billing.calculate(provider, result.tokens),
    });
    return result.text;
  }
}

export class BillingCore {
  charge(tenantId: string, cost: number) {
