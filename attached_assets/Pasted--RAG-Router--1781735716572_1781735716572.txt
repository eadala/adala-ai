دمج النتائج
 ↓
إجابة نهائية قانونية

3. بناء Orchestrator (القلب)

import { contractAgent } from "./agents/contract.js";
import { litigationAgent } from "./agents/litigation.js";
import { riskAgent } from "./agents/risk.js";
import { draftAgent } from "./agents/draft.js";
function routeAgents(prompt) {
  const keywords = {
    contract: ["عقد", "بند", "اتفاقية", "شروط"],
    litigation: ["دعوى", "محكمة", "قضية", "جلسة"],
    risk: ["مخاطر", "مشاكل", "تحذير"],
    draft: ["صياغة", "اكتب", "حرر", "أنشئ"],
  };
  const selected = [];
  if (keywords.contract.some(k => prompt.includes(k))) selected.push("contract");
  if (keywords.litigation.some(k => prompt.includes(k))) selected.push("litigation");
  if (keywords.risk.some(k => prompt.includes(k))) selected.push("risk");
  if (keywords.draft.some(k => prompt.includes(k))) selected.push("draft");
  return selected.length ? selected : ["contract"];
}
export async function orchestrator(prompt, context) {
  const agents = routeAgents(prompt);
  const results = [];
  for (const agent of agents) {
    if (agent === "contract") results.push(await contractAgent(prompt, context));
    if (agent === "litigation") results.push(await litigationAgent(prompt, context));
    if (agent === "risk") results.push(await riskAgent(prompt, context));
    if (agent === "draft") results.push(await draftAgent(prompt, context));
  }
  return results.join("\n\n");
}

4. مثال Agent (العقود)

import axios from "axios";
export async function contractAgent(prompt, context) {
  const res = await axios.post(
    "https://api.openai.com/v1/chat/completions",
    {
      model: "gpt-4o-mini",
      messages: [
        {
          role: "system",
          content: `
أنت محامي متخصص في العقود.
ركز على:
- البنود
- الالتزامات
- الثغرات القانونية
          `,
        },
        {
          role: "user",
          content: `السياق:\n${context}\n\nالسؤال:\n${prompt}`,
        },
      ],
    },
    {
      headers: {
        Authorization: `Bearer ${process.env.OPENAI_API_KEY}`,
      },
    }
  );
  return "📄 تحليل العقود:\n" + res.data.choices[0].message.content;
}

          role: "user",
          content: `

${prompt}
تحليلات الوكلاء:
${results}
          `,
        },
      ],
    },
    {
      headers: {
        Authorization: `Bearer ${process.env.OPENAI_API_KEY}`,
      },
    }
  );

8. الربط النهائي داخل عدالة

import { orchestrator } from "./orch.js";
import { synthesize } from "./synth.js";
export async function legalAI(prompt, context) {
  const agentResults = await orchestrator(prompt, context);
  const finalAnswer = await synthesize(agentResults, prompt);
  return finalAnswer;
}
