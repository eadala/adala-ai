4. AI Router الجديد (احترافي)

import axios from "axios";
import crypto from "crypto";
import Redis from "ioredis";
const redis = new Redis(process.env.REDIS_URL);
// ---------------------
// HASH
// ---------------------
function hash(text) {
  return crypto.createHash("sha256").update(text).digest("hex");
}
// ---------------------
// CLASSIFIER (أذكى)
// ---------------------
function classify(prompt) {
  const p = prompt.length;
  const highValueKeywords = [
    "عقد", "تحليل", "مخاطر", "دعوى", "صياغة", "مراجعة"
  ];
  const isLegal = highValueKeywords.some(k => prompt.includes(k));
  if (p < 80 && !isLegal) return "cheap";
  if (p < 600) return "mid";
  return "premium";
}
// ---------------------
// OPENAI
// ---------------------
async function callOpenAI(prompt) {
  const res = await axios.post(
    "https://api.openai.com/v1/chat/completions",
    {
      model: "gpt-4o-mini",
      messages: [{ role: "user", content: prompt }],
    },
    {
      headers: {
        Authorization: `Bearer ${process.env.OPENAI_API_KEY}`,
      },
    }
  );
  return res.data.choices[0].message.content;
}
// ---------------------
// DEEPSEEK
// ---------------------
async function callDeepSeek(prompt) {
  const res = await axios.post(
    "https://api.deepseek.com/v1/chat/completions",
    {
      model: "deepseek-chat",
      messages: [{ role: "user", content: prompt }],
    },
    {
      headers: {
        Authorization: `Bearer ${process.env.DEEPSEEK_API_KEY}`,
      },
    }
  );
  return res.data.choices[0].message.content;
}
// ---------------------
// USAGE TRACKING
// ---------------------
async function trackUsage(tenantId, level) {
  const costMap = {
    cheap: 1,
    mid: 3,
    premium: 8
  };
  await redis.incrby(`usage:${tenantId}`, costMap[level]);
}
// ---------------------
// MAIN ROUTER
// ---------------------
export async function aiRouter({ tenantId, prompt }) {
  const key = `cache:${tenantId}:${hash(prompt)}`;
  // CACHE CHECK
  const cached = await redis.get(key);
  if (cached) {
    return {
      source: "cache",
      data: JSON.parse(cached),
    };
  }
  // CLASSIFY
  const level = classify(prompt);
  let result;
  if (level === "cheap") {
    result = await callDeepSeek(prompt);
  }
  if (level === "mid") {
    result = await callDeepSeek(prompt);
  }
  if (level === "premium") {
    result = await callOpenAI(prompt);
  }
  // SAVE CACHE
  await redis.set(key, JSON.stringify(result), "EX", 60 * 60 * 24);
  // TRACK USAGE PER TENANT
  await trackUsage(tenantId, level);
  return {
    source: level,
    data: result,
  };
}

5. API Gateway (مع صلاحيات)

app.post("/ai", async (req, res) => {
  const { tenantId, userId, prompt } = req.body;
  // 🔐 حماية بسيطة
  if (!tenantId || !prompt) {
    return res.status(400).json({ error: "missing fields" });
  }
  const result = await aiRouter({ tenantId, prompt });
  res.json({
    tenantId,
    userId,
    ...result,
  });
});
