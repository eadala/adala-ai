✅ الحل: AI TOOL GATEWAY

📦 أنشئ طبقة واحدة فقط:

export const AIDataGateway = {
  getCase: (id, tenantId) =>
    db.execute(sql`
      SELECT id, title, status
      FROM cases
      WHERE id = ${id}
      AND office_id = ${tenantId}
    `),
  listCases: (tenantId) =>
    db.execute(sql`
      SELECT id, title, status
      FROM cases
      WHERE office_id = ${tenantId}
      LIMIT 100
    `),
  getClient: (id, tenantId) =>
    db.execute(sql`
      SELECT id, name
      FROM clients
      WHERE id = ${id}
      AND office_id = ${tenantId}
    `)
};

🔐 4) EXPORT HARD LIMITER (منع Data Dump)

🚫 حماية التصدير

const MAX_EXPORT_LIMIT = 500;
export async function exportCases(req, res) {
  const tenantId = req.officeId;
  const data = await db.execute(sql`
    SELECT * FROM cases
    WHERE office_id = ${tenantId}
    LIMIT ${MAX_EXPORT_LIMIT}
  `);
  return res.json({
    count: data.length,
    data
  });
}

🔎 5) SEARCH SECURITY HARDENING

export async function globalSearch(req, query) {
  const tenantId = req.officeId;
  const safePattern = query.replace(/[%_]/g, "\\$&");
  return db.execute(sql`
    SELECT id, title
    FROM cases
    WHERE office_id = ${tenantId}
    AND title ILIKE ${'%' + safePattern + '%'}
    LIMIT 50
  `);
}
