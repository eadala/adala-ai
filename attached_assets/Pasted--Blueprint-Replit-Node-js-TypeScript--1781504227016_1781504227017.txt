3. ⚡ Stripe Webhook (Stateless Entry Point)

// src/api/routes/stripe/webhook.ts
import express from "express";
import { enqueueStripeEvent } from "@/core/queue/queue";
const router = express.Router();
router.post("/", express.raw({ type: "application/json" }), async (req, res) => {
  const signature = req.headers["stripe-signature"];
  if (!signature) {
    return res.status(400).json({ error: "Missing signature" });
  }
  try {
    await enqueueStripeEvent({
      body: req.body,
      signature: signature as string,
    });
    return res.status(200).json({ received: true });
  } catch (err) {
    return res.status(400).json({ error: "Webhook rejected" });
  }
});
export default router;

6. 🧠 Stripe Processor (Core Logic)

// src/core/services/stripe/stripe.processor.ts
import { db } from "@/core/db/client";
import { stripe } from "@/core/config/stripe";
export async function processStripeEvent({ body, signature }) {
  const rawEvent = JSON.parse(body.toString());
  // 1. Idempotency Check
  const exists = await db.stripe_events.findUnique({
    where: { stripe_event_id: rawEvent.id },
  });
  if (exists) return;
  // 2. Save Event First (Safety Buffer)
  await db.stripe_events.create({
    data: {
      stripe_event_id: rawEvent.id,
      type: rawEvent.type,
      payload: rawEvent,
      status: "processing",
    },
  });
  try {
    // 3. Verify Signature (CRITICAL)
    const event = stripe.webhooks.constructEvent(
      body,
      signature,
      process.env.STRIPE_WEBHOOK_SECRET!
    );
    // 4. Route event
    await handleEvent(event);
    // 5. Mark Done
    await db.stripe_events.update({
      where: { stripe_event_id: event.id },
      data: { status: "done" },
    });
  } catch (err) {
    await db.stripe_events.update({
      where: { stripe_event_id: rawEvent.id },
      data: {
        status: "failed",
        last_error: String(err),
      },
    });
    throw err;
  }
}

8. 🔁 Reconciliation Job (Critical for Finance)

// src/jobs/stripe.reconcile.job.ts
export async function reconcileStripe() {
  const stripePayments = await stripe.paymentIntents.list();
  const dbPayments = await db.ledger.findMany();
  // detect mismatch
  const missing = stripePayments.data.filter(
    p => !dbPayments.find(d => d.reference === p.id)
  );
  if (missing.length > 0) {
    console.warn("MISMATCH DETECTED", missing);
  }
}
