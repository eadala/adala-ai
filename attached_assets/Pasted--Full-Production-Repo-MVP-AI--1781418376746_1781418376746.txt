🧠 3) Orchestration Engine (Core MVP Logic)

@Injectable()
export class OrchestrationService {
  constructor(
    private deals: DealsService,
    private cases: CasesService,
    private contracts: ContractsService
  ) {}
  async onPaymentSuccess(order: any) {
    const deal = await this.deals.createFromOrder(order)
    const contract = await this.contracts.generate(deal.id)
    const caseEntity = await this.cases.createFromContract(contract)
    return {
      deal,
      contract,
      caseEntity
    }
  }
}

📦 5) Orders Module

@Controller('orders')
export class OrdersController {
  constructor(
    private orders: OrdersService,
    private orchestration: OrchestrationService
  ) {}
  @Post()
  create(@Body() dto: any) {
    return this.orders.create(dto)
  }
  @Post(':id/pay')
  async pay(@Param('id') id: string) {
    const order = await this.orders.markPaid(id)
    return this.orchestration.onPaymentSuccess(order)
  }
}

📄 7) Contract Generator

@Injectable()
export class ContractsService {
  constructor(private prisma: PrismaService) {}
  async generate(dealId: string) {
    const deal = await this.prisma.deals.findUnique({ where: { id: dealId } })
    return this.prisma.contracts.create({
      data: {
        deal_id: dealId,
        content: {
          title: "Legal Service Agreement",
          client: deal.client_id,
          price: deal.final_price
        },
        status: "draft"
      }
    })
  }
}

⚖️ 8) Case System

@Injectable()
export class CasesService {
  constructor(private prisma: PrismaService) {}
  async createFromContract(contract: any) {
    return this.prisma.cases.create({
      data: {
        contract_id: contract.id,
        client_id: contract.client_id,
        status: "active"
      }
    })
  }
}

export default async function Marketplace() {
  const res = await fetch('http://localhost:3001/services')
  const services = await res.json()
  return (
    <div>
      <h1>⚖️ الخدمات القانونية</h1>

        <a key={s.id} href={`/service/${s.id}`}>
          <div>
            <h2>{s.title}</h2>
            <p>{s.price} SAR</p>
          </div>
        </a>
      ))}
    </div>
  )
}
