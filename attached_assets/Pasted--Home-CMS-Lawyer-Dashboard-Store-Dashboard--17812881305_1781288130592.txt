export async function aiRouter(type: string, input: string) {
  switch (type) {
    case "marketing":
      return generateMarketingText(input);
    case "legal_case":
      return analyzeCase(input);
    case "contract":
      return generateContract(input);
    case "store":
      return generateProductDescription(input);
    default:
      return generalAI(input);
  }
}

🔐 Middleware

export function auth(role) {
  return (req, res, next) => {
    if (!req.user) return res.sendStatus(401);
    if (req.user.role !== role && req.user.role !== "OWNER") {
      return res.sendStatus(403);
    }
    next();
  };
}
