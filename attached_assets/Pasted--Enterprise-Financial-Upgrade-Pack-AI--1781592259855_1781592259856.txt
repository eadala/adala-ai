🧠 3) DOUBLE-ENTRY LEDGER (القلب الجديد للنظام)

📦 الجدول الأساسي:

CREATE TABLE ledger_entries (
  id TEXT PRIMARY KEY,
  office_id TEXT NOT NULL,
  entry_type TEXT CHECK(entry_type IN ('DEBIT','CREDIT')),
  amount DECIMAL(12,2) NOT NULL,
  currency TEXT DEFAULT 'SAR',
  reference_type TEXT, -- invoice, payment, case, expense
  reference_id TEXT,
  description TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

⸻

🔴 قاعدة محاسبية إلزامية:

كل عملية مالية = Debit + Credit متوازن

⸻

مثال:

إنشاء فاتورة:

Debit  : Accounts Receivable (Client)
Credit : Revenue

⸻

استلام دفعة:

Debit  : Cash
Credit : Accounts Receivable

⸻

🔄 4) FINANCIAL EVENT ENGINE

كل حدث مالي يتحول تلقائيًا إلى قيود دفترية:

export function recordFinancialEvent(event) {
  switch (event.type) {
    case "INVOICE_CREATED":
      createLedgerEntry({
        debit: "AR",
        credit: "REVENUE",
        amount: event.amount
      });
      break;
    case "PAYMENT_RECEIVED":
      createLedgerEntry({
        debit: "CASH",
        credit: "AR",
        amount: event.amount
      });
      break;
    case "EXPENSE":
      createLedgerEntry({
        debit: "EXPENSE",
        credit: "CASH",
        amount: event.amount
      });
      break;
  }
}

🤖 8) AI FINANCIAL GUARD

يمنع AI من التلاعب أو القراءة الخام

export function aiFinanceGuard(data, tenantId) {
  return data.map(entry => {
    if (entry.office_id !== tenantId) {
      throw new Error("CROSS_TENANT_BLOCKED");
    }
    return {
      type: entry.type,
      amount: entry.amount,
      safe: true
    };
  });
}
