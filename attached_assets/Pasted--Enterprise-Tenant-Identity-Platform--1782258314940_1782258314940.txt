⚙️ الكود

export type TenantResolutionTrace = {
  tenantId: string;
  role: string;
  source: "office_members" | "registry";
  steps: string[];
};
export async function resolveTenantWithTrace(userId: string): Promise<TenantResolutionTrace> {
  const steps: string[] = [];
  // 1. المصدر الأساسي
  const member = await db.query.office_members.findFirst({
    where: (t, { eq }) => eq(t.userId, userId)
  });
  if (member) {
    steps.push("FOUND_IN_office_members");
    return {
      tenantId: member.officeId,
      role: member.role,
      source: "office_members",
      steps
    };
  }
  steps.push("NOT_FOUND_IN_office_members");
  // 2. registry fallback controlled (ONBOARDING ONLY)
  const registry = await db.query.office_registry.findFirst({
    where: (t, { eq }) => eq(t.userId, userId)
  });
  if (registry) {
    steps.push("FOUND_IN_registry");
    const created = await db.insert(office_members).values({
      userId,
      officeId: registry.officeId,
      role: registry.role ?? "member"
    }).returning();
    steps.push("AUTO_LINKED_TO_office_members");
    return {
      tenantId: created[0].officeId,
      role: created[0].role,
      source: "registry",
      steps
    };
  }
  steps.push("TENANT_RESOLUTION_FAILED");
  throw new Error("TENANT_NOT_RESOLVED");
}

export async function logTenantResolution(userId: string, trace: any) {
  await db.insert(tenant_audit_logs).values({
    userId,
    tenantId: trace.tenantId ?? null,
    source: trace.source ?? "none",
    steps: trace.steps,
    createdAt: new Date()
  });
}

export async function debugTenant(req, res) {
  const userId = req.params.userId;
  const trace = await resolveTenantWithTrace(userId);
  return res.json({
    status: "ok",
    tenant: trace.tenantId,
    role: trace.role,
    resolutionPath: trace.steps
  });
}

🧭 4️⃣ Tenant Switch (بدون Logout)

⚠️ مهم: للمطور + مالك النظام فقط

POST /dev/tenant/switch
export async function switchTenant(req, res) {
  const { targetTenantId } = req.body;
  if (req.user.role !== "developer" && req.user.role !== "firm_owner") {
    return res.status(403).json({ error: "NOT_ALLOWED" });
  }
  req.session.activeTenant = targetTenantId;
  return res.json({
    success: true,
    activeTenant: targetTenantId
  });
}

🧠 5️⃣ Tenant Context Resolver (Final Middleware)

export async function requireTenant(req, res, next) {
  try {
    const userId = req.user.id;
    const trace = await resolveTenantWithTrace(userId);
    await logTenantResolution(userId, trace);
    req.tenantId = trace.tenantId;
    req.role = trace.role;
    req.tenantTrace = trace.steps;
    return next();
  } catch (err) {
    return res.status(403).json({
      error: "TENANT_NOT_RESOLVED",
      code: "TNT_403"
    });
  }
}
