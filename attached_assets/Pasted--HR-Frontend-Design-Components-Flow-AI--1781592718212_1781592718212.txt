🧠 3) ROLE & PERMISSION MATRIX (قلب النظام)

export function RolesPage({ roles, permissions }) {
  return (
    <div>
      <h2 className="text-2xl font-bold mb-6">Roles & Permissions</h2>
      <div className="grid grid-cols-3 gap-4">
        
        {roles.map(role => (
          <div className="bg-white p-4 rounded-xl border">
            <h3 className="font-bold mb-2">{role.name}</h3>
            <div className="space-y-2">
              {permissions.map(p => (
                <label className="flex items-center gap-2 text-sm">
                  <input type="checkbox" checked={role.permissions.includes(p.key)} />
                  {p.key}
                </label>
              ))}
            </div>
            <button className="mt-4 w-full bg-black text-white py-2 rounded">
              Save
            </button>
          </div>
        ))}
      </div>
    </div>
  );
}

🔐 7) HR GUARD UI (تحذيرات الصلاحيات)

export function PermissionWarning({ user }) {
  if (!user.hasPermission) {
    return (
      <div className="bg-red-100 text-red-700 p-4 rounded-xl">
        ⚠️ You do not have permission to access this module
      </div>
    );
  }
  return null;
}
