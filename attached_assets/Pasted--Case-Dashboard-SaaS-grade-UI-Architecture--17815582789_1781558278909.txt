🧱 2. CASE PAGE (ROOT COMPONENT)

export default function CasePage({ caseId }) {
  const { data: caseData } = useCase(caseId);
  return (
    <div className="case-container">
      <CaseHeader case={caseData} />
      <CaseActionBar case={caseData} />
      <div className="case-grid">
        <CaseTimeline caseId={caseId} />
        <CaseSidebar case={caseData} />
      </div>
      <CaseActivityFeed caseId={caseId} />
    </div>
  );
}

🧠 3. HEADER (هوية القضية)

function CaseHeader({ case }) {
  return (
    <div className="case-header">
      <div>
        <h1>{case.title}</h1>
        <p>Case ID: {case.id}</p>
      </div>
      <div className="case-meta">
        <span>{case.type}</span>
        <span>{case.status}</span>
      </div>
    </div>
  );
}

🎛️ 4. ACTION BAR (الإجراءات السريعة)

function CaseActionBar({ case }) {
  return (
    <div className="action-bar">
      <button>➕ مهمة</button>
      <button>💬 مراسلة</button>
      <button>📄 رفع مستند</button>
      <button>📅 إضافة جلسة</button>

⸻

🕒 5. TIMELINE (قلب النظام)

function CaseTimeline({ caseId }) {
  const { data } = useCaseTimeline(caseId);
  return (
    <div className="timeline">
      {data.map(event => (
        <div className="timeline-item" key={event.id}>
          <span>{event.type}</span>
          <p>{event.description}</p>
          <small>{event.timestamp}</small>
        </div>
      ))}
    </div>
  );
}

⸻

💬 6. COMMUNICATION PANEL (مراسلات داخل القضية)

function CaseMessages({ caseId }) {
  const { data } = useCaseMessages(caseId);
  return (
    <div className="messages">
      {data.map(msg => (
        <div className={`msg ${msg.type}`}>
          <strong>{msg.sender}</strong>
          <p>{msg.message}</p>
        </div>
      ))}
      <input placeholder="اكتب رسالة..." />
    </div>
  );
}

⸻

📋 7. TASK BOARD (مهام القضية)

function CaseTasks({ caseId }) {
  const { data } = useCaseTasks(caseId);
  return (
    <div className="tasks">
      <h3>المهام</h3>
      {data.map(task => (
        <div className={`task ${task.status}`}>
          <p>{task.title}</p>
          <span>{task.due_date}</span>
        </div>
      ))}
    </div>
  );
}

⸻

📄 8. DOCUMENT PANEL

function CaseDocuments({ caseId }) {
  const { data } = useCaseDocuments(caseId);
  return (
    <div className="documents">
      <h3>المستندات</h3>
      {data.map(doc => (
        <div className="doc">
          <a href={doc.file_url}>📄 فتح الملف</a>
        </div>
      ))}
    </div>
  );
}

📡 10. ACTIVITY FEED (أسفل الصفحة)

function CaseActivityFeed({ caseId }) {
  const { data } = useCaseActivity(caseId);
  return (
    <div className="activity-feed">
      {data.map(event => (
        <div className="activity">
          <span>{event.type}</span>
          <p>{event.text}</p>
        </div>
      ))}
    </div>
  );
}

🧠 12. DESIGN PRINCIPLES

واجهة القضية يجب أن تكون:

* 🧭 مركزية (Everything in one place)
* ⚡ سريعة (no navigation overload)
* 📊 تحليلية (AI insights)
* 🧾 تشغيلية (tasks + docs + messages)
* 🧠 سياقية (everything linked to case_id)
