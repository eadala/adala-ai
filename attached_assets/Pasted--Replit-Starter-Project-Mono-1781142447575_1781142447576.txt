app.use("/api/events", eventRoutes);
app.get("/health", (req, res) => {
  res.json({
    status: "OK",
    system: "Adalah AI V2 (Replit Edition)"
  });
});
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log("Adalah AI running on port", PORT);
});

🧠 5. AI Engine (قلب النظام)

src/core/aiEngine.js

export const aiEngine = {
  async summarizeCase(caseData) {
    return {
      summary: "تحليل أولي للقضية بناء على البيانات الحالية",
      risks: [
        "تأخير في المستندات",
        "احتمال ضعف في الإثبات"
      ],
      recommendations: [
        "تعزيز الأدلة",
        "إعادة ترتيب الدفوع"
      ],
      opponentStrategy: "متوقع الدفع ببطلان أو نقص الإثبات"
    };
  },
  async analyzeEvent(event) {
    switch (event.type) {
      case "DOCUMENT_UPLOADED":
        return {
          insight: "تم استخراج أطراف وتواريخ من المستند",
          action: "index_document"
        };
      case "CASE_UPDATED":
        return {
          insight: "تحديث حالة القضية يؤثر على التقييم العام",
          action: "recalculate_risk"
        };
      case "INVOICE_CREATED":
        return {
          insight: "ربط مالي جديد بالقضية",
          action: "link_financial_model"
        };
      default:
        return { insight: "no_action" };
    }
  }
};

⚡ 6. Event Bus (مهم جدًا)

src/core/eventBus.js

import { PrismaClient } from "@prisma/client";
import { aiEngine } from "./aiEngine.js";
const prisma = new PrismaClient();
export async function emitEvent(event) {
  const saved = await prisma.event.create({
    data: event
  });
  const aiResult = await aiEngine.analyzeEvent(event);
  return {
    event: saved,
    ai: aiResult
  };
}

📁 7. Case Routes

src/routes/cases.js

import express from "express";
import { PrismaClient } from "@prisma/client";
import { aiEngine } from "../core/aiEngine.js";
const router = express.Router();
const prisma = new PrismaClient();
// Create case
router.post("/", async (req, res) => {
  const caseData = await prisma.case.create({
    data: req.body
  });
  res.json(caseData);
});
// Case Hub (CORE ENDPOINT)
router.get("/:id/hub", async (req, res) => {
  const caseData = await prisma.case.findUnique({
    where: { id: req.params.id }
  });
  const ai = await aiEngine.summarizeCase(caseData);
  res.json({
    case: caseData,
    ai
  });
});
export default router;

👤 8. Clients 360

src/routes/clients.js

import express from "express";
import { PrismaClient } from "@prisma/client";
const router = express.Router();
const prisma = new PrismaClient();
// Create client
router.post("/", async (req, res) => {
  const client = await prisma.client.create({
    data: req.body
  });
  res.json(client);
});
// Client 360
router.get("/:id/overview", async (req, res) => {
  const client = await prisma.client.findUnique({
    where: { id: req.params.id }
  });
  const cases = await prisma.case.findMany({
    where: { clientId: req.params.id }
  });
  res.json({
    client,
    cases
  });
});
export default router;
