  const tokens = {
    colors: figmaData.colors,
    spacing: figmaData.spacing,

  };
  await saveToProject("/tokens/design-tokens.json", tokens);

🧩 5) Component Mapper (Figma → React)

📁 tools/component-mapper.ts

export function mapComponent(figmaNode: any) {
  switch (figmaNode.type) {
    case "BUTTON":
      return `
        <Button variant="${figmaNode.variant}">
          ${figmaNode.text}
        </Button>
      `;
    case "CARD":
      return `
        <Card>
          ${figmaNode.children}
        </Card>
      `;
    case "INPUT":
      return `<Input placeholder="${figmaNode.placeholder}" />`;
    default:
      return `<div>${figmaNode.children}</div>`;
  }
}

⚡ 6) Code Generator Engine

📁 tools/code-generator.ts

export function generateReact(componentTree: any) {
  return componentTree.map((node: any) => {
    if (node.type === "CARD") {
      return `
        <Card>
          ${generateReact(node.children)}
        </Card>
      `;
    }
    if (node.type === "TEXT") {
      return `<p>${node.content}</p>`;
    }
    return "";
  }).join("\n");
}
