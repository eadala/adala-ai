// admin/BillingDashboard.tsx
import { useState } from "react";
export default function BillingDashboard() {
  const [revenue] = useState(114000);
  const [cost] = useState(24000);
  const [customers] = useState(120);
  const [aiUsage] = useState(8500000);
  const profit = revenue - cost;
  const margin = ((profit / revenue) * 100).toFixed(1);
  return (
    <div className="p-6 grid gap-6">
      <h1 className="text-2xl font-bold">عدالة AI — لوحة الإيرادات</h1>
      {/* KPIs */}
      <div className="grid grid-cols-4 gap-4">
        <Card title="الإيرادات الشهرية" value={`${revenue} SAR`} />
        <Card title="التكلفة التشغيلية" value={`${cost} SAR`} />
        <Card title="صافي الربح" value={`${profit} SAR`} />
        <Card title="هامش الربح" value={`${margin}%`} />
      </div>
      {/* Usage */}
      <div className="grid grid-cols-2 gap-4">
        <Card title="عدد العملاء" value={customers} />
        <Card title="استهلاك AI" value={aiUsage.toLocaleString()} />
      </div>
      {/* Chart Placeholder */}
      <div className="h-64 bg-gray-100 rounded-xl flex items-center justify-center">
        Revenue Growth Chart (Monthly)
      </div>
    </div>
  );
}
function Card({ title, value }: any) {
  return (
    <div className="p-4 rounded-xl bg-white shadow">
      <div className="text-sm text-gray-500">{title}</div>
      <div className="text-xl font-bold">{value}</div>
    </div>
  );
}

model Invoice {

💳 4) Stripe Integration (Backend-ready)

import Stripe from "stripe";
export const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: "2024-06-20",
});
export async function createCheckoutSession(plan: string) {
  return await stripe.checkout.sessions.create({
    payment_method_types: ["card"],
    mode: "subscription",
    line_items: [
      {
        price_data: {
          currency: "sar",
          product_data: {
            name: `عدالة AI - ${plan}`,
          },
          unit_amount: 19900,
          recurring: {
            interval: "month",
          },
        },
        quantity: 1,
      },
    ],
    success_url: `${process.env.APP_URL}/success`,
    cancel_url: `${process.env.APP_URL}/cancel`,
  });
}

📈 5) KPI Logic (Real SaaS Metrics)

export function calculateKPIs(revenue: number, cost: number, customers: number) {
  return {
    revenue,
    cost,
    profit: revenue - cost,
    margin: ((revenue - cost) / revenue) * 100,
    arpu: revenue / customers,
    ltv: (revenue / customers) * 12,
  };
}
