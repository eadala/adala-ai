🔐 3. طبقة الحماية (Role Middleware) — أهم جزء

import { NextResponse } from "next/server";
import type { NextRequest } from "next/server";
export function middleware(req: NextRequest) {
  const token = req.cookies.get("token")?.value;
  const role = req.cookies.get("role")?.value;
  const path = req.nextUrl.pathname;
  // حماية لوحة الأدمن
  if (path.startsWith("/admin")) {
    if (role !== "platform_admin") {
      return NextResponse.redirect(new URL("/not-authorized", req.url));
    }
  }
  // حماية مساحة المكتب القانوني
  if (path.startsWith("/workspace")) {
    if (!["law_firm_owner", "lawyer"].includes(role || "")) {
      return NextResponse.redirect(new URL("/not-authorized", req.url));
    }
  }
  // حماية العميل
  if (path.startsWith("/client")) {
    if (role !== "client") {
      return NextResponse.redirect(new URL("/not-authorized", req.url));
    }
  }
  return NextResponse.next();
}

🧭 4. نظام توجيه ذكي بعد تسجيل الدخول

بدل ما كل المستخدمين يذهبون لنفس الصفحة:

export function redirectByRole(role: string) {
  switch (role) {
    case "platform_admin":
      return "/admin";
    case "law_firm_owner":
    case "lawyer":
      return "/workspace";
    case "client":
      return "/client";
    default:
      return "/login";
  }
}
