6. تحويل النص إلى Embedding

import axios from "axios";
export async function createEmbedding(text) {
  const res = await axios.post(
    "https://api.openai.com/v1/embeddings",
    {
      model: "text-embedding-3-small",
      input: text,
    },
    {
      headers: {
        Authorization: `Bearer ${process.env.OPENAI_API_KEY}`,
      },
    }
  );
  return res.data.data[0].embedding;
}

7. حفظ المستند (Chunking + Storage)

import { createEmbedding } from "./embed.js";
import pg from "pg";
const db = new pg.Pool({
  connectionString: process.env.DATABASE_URL,
});
function splitText(text, size = 500) {
  const chunks = [];
  for (let i = 0; i < text.length; i += size) {
    chunks.push(text.slice(i, i + size));
  }
  return chunks;
}
export async function storeDocument(tenantId, text) {
  const chunks = splitText(text);
  for (const chunk of chunks) {
    const embedding = await createEmbedding(chunk);
    await db.query(
      `INSERT INTO documents (tenant_id, content, embedding)
       VALUES ($1, $2, $3)`,
      [tenantId, chunk, embedding]
    );
  }
}

8. البحث الذكي (RAG Retrieval)

import { createEmbedding } from "./embed.js";
import pg from "pg";
const db = new pg.Pool({
  connectionString: process.env.DATABASE_URL,
});
export async function searchDocuments(tenantId, query) {
  const embedding = await createEmbedding(query);
  const result = await db.query(
    `
    SELECT content
    FROM documents
    WHERE tenant_id = $1
    ORDER BY embedding <-> $2
    LIMIT 5
    `,
    [tenantId, embedding]
  );
  return result.rows.map(r => r.content).join("\n");
}

9. دمج RAG مع AI Router

import { searchDocuments } from "./rag_search.js";
import { callOpenAI } from "./ai.js";
export async function legalAI({ tenantId, prompt }) {
  // 1. جلب المعرفة الداخلية
  const context = await searchDocuments(tenantId, prompt);
  // 2. بناء prompt ذكي
  const finalPrompt = `
أنت مساعد قانوني داخل مكتب.
اعتمد فقط على المعلومات التالية:
--- المستندات ---
${context}
--- السؤال ---
${prompt}
أعط إجابة قانونية دقيقة.
`;
  // 3. إرسال للنموذج
  return await callOpenAI(finalPrompt);
}
