🧠 2. حساب الأداء (Performance Engine)

function calculatePerformance(user) {
  let score = 100;
  score += user.casesClosed * 5;
  score -= user.casesDelayed * 8;
  score += user.tasksCompleted * 2;
  score -= user.errors * 7;
  score += user.onTimeDays * 3;
  score -= user.lateDays * 5;
  score -= user.absentDays * 10;
  return Math.max(0, Math.min(100, score));
}

⸻

💰 3. نظام الرواتب الذكي

🎯 الفكرة:

الراتب = أساسي + حافز أداء - خصومات

⸻

function calculateSalary(user) {
  const performance = calculatePerformance(user);
  let bonus = 0;
  let deduction = 0;
  // 🔥 الحوافز
  if (performance >= 90) bonus = user.baseSalary * 0.30;
  else if (performance >= 80) bonus = user.baseSalary * 0.20;
  else if (performance >= 70) bonus = user.baseSalary * 0.10;
  // ⚠️ الخصومات
  if (user.lateDays > 5) deduction += user.baseSalary * 0.05;
  if (user.absentDays > 2) deduction += user.baseSalary * 0.10;
  if (performance < 60) deduction += user.baseSalary * 0.15;
  const finalSalary = user.baseSalary + bonus - deduction;
  return {
    baseSalary: user.baseSalary,
    performance,
    bonus,
    deduction,
    finalSalary: Math.max(0, finalSalary)
  };
}
