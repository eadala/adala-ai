⚙️ Orchestration Service

@Injectable()
export class OrchestrationService {
  constructor(
    private deals: DealsService,
    private cases: CasesService,
    private contracts: ContractsService,
  ) {}
  async handleOrderPaid(order: Order) {
    const deal = await this.deals.createFromOrder(order)
    if (order.service.type === 'fixed') {
      return this.createContractFlow(deal.id)
    }
    return deal
  }
  async createContractFlow(dealId: string) {
    const contract = await this.contracts.generate(dealId)
    await this.cases.createFromContract(contract)
    return contract
  }
}

@Injectable()
export class MarketplaceService {
  constructor(private prisma: PrismaService) {}
  findAll(tenantId: string) {
    return this.prisma.services.findMany({
      where: { tenant_id: tenantId }
    })
  }
  create(dto: CreateServiceDto) {
    return this.prisma.services.create({ data: dto })
  }
}

@Controller('orders')

  @Post()

    return this.orders.create(dto)
  }

📄 7) Contracts Module

@Injectable()
export class ContractsService {
  constructor(private prisma: PrismaService) {}
  async generate(dealId: string) {
    const deal = await this.prisma.deals.findUnique({
      where: { id: dealId }
    })
    return this.prisma.contracts.create({
      data: {
        deal_id: dealId,
        content: this.buildTemplate(deal),
        status: 'draft'
      }
    })
  }
  buildTemplate(deal: any) {
    return {
      client: deal.client_id,
      price: deal.final_price,
      terms: "Legal Service Agreement"
    }
  }
}

🔐 9) RBAC Guard (مهم جداً)

@Injectable()
export class RbacGuard implements CanActivate {
  canActivate(context: ExecutionContext): boolean {
    const user = context.switchToHttp().getRequest().user
    const action = context.getHandler().name
    return user.permissions.includes(action)
  }
}
