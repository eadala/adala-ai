🔌 2) Provider System (مثال OpenAI)

export class OpenAIProvider implements AIProvider {
  async generate(prompt: string) {
    const res = await fetch("https://api.openai.com/v1/chat/completions", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${process.env.OPENAI_KEY}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        model: "gpt-4o-mini",
        messages: [{ role: "user", content: prompt }]
      })
    });
    const data = await res.json();
    return {
      text: data.choices[0].message.content,
      tokens: data.usage.total_tokens
    };
  }
}

🧠 3) AI Router (المخ الأساسي)

export class AIRouter {
  constructor(private providers: Record<string, AIProvider>) {}
  async run(provider: string, prompt: string) {
    const selected = this.providers[provider];
    if (!selected) throw new Error("Provider not found");
    return selected.generate(prompt);
  }
}

🧠 4) Policy Engine (الذكاء الحقيقي للنظام)

export class PolicyEngine {
  selectProvider(input: any) {
    if (input.type === "legal_analysis") return "claude";
    if (input.cost_sensitive) return "gemini";
    if (input.fast_mode) return "gemini_flash";
    return "openai";
  }
}

🧮 حساب التكلفة

export class CostCalculator {
  calculate(provider: string, tokens: number) {
    const rates = {
      openai: 0.002,
      gemini: 0.0005,
      claude: 0.003
    };
    return tokens * rates[provider];
  }
}

🧠 Legal Prompt Engine

export class LegalAI {
  buildPrompt(type: string, data: any) {
    switch(type) {
      case "contract":
        return `اكتب عقد قانوني سعودي مطابق للأنظمة: ${data}`;
      case "lawsuit":
        return `صغ لائحة دعوى احترافية: ${data}`;
      case "analysis":
        return `حلل هذا العقد قانونياً: ${data}`;
    }
  }
}
