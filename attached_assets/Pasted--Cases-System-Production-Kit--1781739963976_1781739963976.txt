app.post("/cases", async (req, res) => {
  const ctx = req.context;
  const {
    caseNumber,
    title,
    type,
    court_name,
    court_code,
    district_number,
    district_type,
    branchId
  } = req.body;
  const result = await db.query(
    `INSERT INTO cases
    (id, tenant_id, branch_id, case_number, title, type,
     court_name, court_code, court_district_number, court_district_type)
     VALUES (gen_random_uuid(), $1,$2,$3,$4,$5,$6,$7,$8,$9)
     RETURNING *`,
    [
      ctx.tenantId,
      branchId,
      caseNumber,
      title,
      type,
      court_name,
      court_code,
      district_number,
      district_type
    ]
  );
  res.json(result.rows[0]);
});

  const { id } = req.params;
  const { hearing_date, court_room, notes } = req.body;
  await db.query(
    `INSERT INTO hearings (id, case_id, hearing_date, court_room, notes)
     VALUES (gen_random_uuid(), $1,$2,$3,$4)`,
    [id, hearing_date, court_room, notes]
  );
  await syncNextHearing(id);

🔁 تحديث الجلسة القادمة تلقائياً

async function syncNextHearing(caseId) {
  const res = await db.query(
    `SELECT * FROM hearings
     WHERE case_id=$1 AND hearing_date > NOW()
     ORDER BY hearing_date ASC LIMIT 1`,
    [caseId]
  );
  const next = res.rows[0];
  await db.query(
    `UPDATE cases SET next_hearing_date=$1 WHERE id=$2`,
    [next?.hearing_date || null, caseId]
  );
}

function checkAlerts(date) {
  const diff = new Date(date) - new Date();
  if (diff < 7 * 86400000) return "7_days";
  if (diff < 3 * 86400000) return "3_days";
  if (diff < 86400000) return "24_hours";
}
