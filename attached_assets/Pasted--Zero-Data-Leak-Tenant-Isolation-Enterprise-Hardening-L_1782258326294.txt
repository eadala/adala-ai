⸻

⚙️ الكود الأساسي

export function enforceTenantIsolation(query: any, tenantId: string) {
  if (!tenantId) {
    throw new Error("TENANT_MISSING");
  }
  // 🛡️ منع أي query بدون tenant filter
  if (!query.where || !query.where.tenantId) {
    query.where = {
      ...query.where,
      tenantId: tenantId
    };
  }
  return query;
}

🧠 4️⃣ DB Guard Layer (Drizzle Middleware)

export function tenantSafeQuery(db, tenantId: string) {
  return {
    query: (table, builderFn) => {
      return db.query(table, (q, helpers) => {
        const base = builderFn(q, helpers);
        // 🛡️ FORCE injection
        return {
          ...base,
          where: (t, ops) =>
            ops.and(
              base.where ? base.where(t, ops) : undefined,
              ops.eq(t.tenantId, tenantId)
            )
        };
      });
    }
  };
}

🔒 5️⃣ Middleware إجباري (Hard Gate)

export function requireStrictTenant(req, res, next) {
  if (!req.tenantId) {
    return res.status(403).json({
      error: "ISOLATION_FAILURE"
    });
  }
  // 🛡️ منع أي override يدوي
  req.body.tenantId = req.tenantId;
  req.query.tenantId = req.tenantId;
  next();
}

CREATE POLICY tenant_isolation ON cases
USING (tenant_id = current_setting('app.tenant_id')::text);
