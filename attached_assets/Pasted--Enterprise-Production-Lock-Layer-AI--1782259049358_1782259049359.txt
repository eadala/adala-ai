🔐 1️⃣ Tenant Versioning System (حل Drift النهائي)

📌 المشكلة التي يحلها

* تغير المكتب في trial_registry
* أو إعادة onboarding
* أو إعادة تعيين المستخدم

➡️ بدون هذا: الهوية “تتجمد بشكل خاطئ”

⸻

⚙️ الكود

export type TenantBinding = {
  userId: string;
  tenantId: string;
  version: number;
  source: "office_members" | "registry" | "trial";
  updatedAt: Date;
};
export async function bindTenant(userId: string, tenantId: string, source: string) {
  const existing = await db.query.tenant_bindings.findFirst({
    where: (t, { eq }) => eq(t.userId, userId)
  });
  // 🧠 إذا تغيرت الهوية → نرفع version
  if (existing && existing.tenantId !== tenantId) {
    await db.insert(tenant_binding_history).values({
      ...existing,
      endedAt: new Date()
    });
    return await db.update(tenant_bindings)
      .set({
        tenantId,
        version: existing.version + 1,
        source,
        updatedAt: new Date()
      })
      .where(eq(tenant_bindings.userId, userId));
  }
  // 🆕 أول مرة
  return await db.insert(tenant_bindings).values({
    userId,
    tenantId,
    version: 1,
    source,
    updatedAt: new Date()
  });
}

بدلاً من تخزين كل شيء raw → يتم ضغطه دورياً

⸻

⚙️ النظام

export async function compressAuditLogs() {
  const logs = await db.query.tenant_audit_logs.findMany({
    where: (t, { lt }) =>
      lt(t.createdAt, new Date(Date.now() - 7 * 24 * 60 * 60 * 1000))
  });
  const grouped = groupBy(logs, "tenantId");
  for (const tenantId in grouped) {
    await db.insert(tenant_audit_archive).values({
      tenantId,
      summary: {
        total: grouped[tenantId].length,
        failures: grouped[tenantId].filter(l => l.status === "fail").length,
        sources: [...new Set(grouped[tenantId].map(l => l.source))]
      }
    });
    await db.delete(tenant_audit_logs)
      .where(eq(tenant_audit_logs.tenantId, tenantId));
  }
}

⚙️ الكود

export async function requireAdminShield(req, res, next) {
  if (req.user.role !== "developer") {
    return next();
  }
  // 🧠 2FA إجباري
  if (!req.session?.twoFactorVerified) {
    return res.status(403).json({
      error: "2FA_REQUIRED"
    });
  }
  // 🔐 session binding check
  if (req.session.ip !== req.ip) {
    return res.status(403).json({
      error: "SESSION_MISMATCH"
    });
  }
  return next();
}

🧯 4️⃣ Disaster Identity Recovery (أهم جزء)

📌 يستخدم عند:

* corruption في tenant_bindings
* حذف بيانات بالخطأ
* bug في onboarding

⸻

⚙️ النظام

export async function recoverIdentity(userId: string) {
  const history = await db.query.tenant_binding_history.findMany({
    where: (t, { eq }) => eq(t.userId, userId),
    orderBy: (t, { desc }) => desc(t.version)
  });
  const lastValid = history[0];
  if (!lastValid) throw new Error("NO_RECOVERY_POINT");
  await db.update(tenant_bindings)
    .set({
      tenantId: lastValid.tenantId,
      version: lastValid.version,
      source: "recovered",
      updatedAt: new Date()
    })
    .where(eq(tenant_bindings.userId, userId));
}
