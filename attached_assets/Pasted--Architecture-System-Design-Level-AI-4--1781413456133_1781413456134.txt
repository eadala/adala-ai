3) 🗄️ Database Triggers Strategy (PostgreSQL)

بدل polling، نستخدم triggers لإطلاق الأحداث فورياً.

⸻

📌 مثال Trigger: Invoice Overdue

CREATE OR REPLACE FUNCTION notify_invoice_overdue()
RETURNS trigger AS $$
BEGIN
  IF NEW.due_date < NOW() AND NEW.status != 'paid' THEN
    INSERT INTO ai_events(type, payload)
    VALUES (
      'INVOICE_OVERDUE',
      json_build_object(
        'invoice_id', NEW.id,
        'client_id', NEW.client_id,
        'amount', NEW.amount
      )
    );
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

⸻

📌 Trigger Binding

CREATE TRIGGER invoice_overdue_trigger
AFTER UPDATE ON invoices
FOR EACH ROW
EXECUTE FUNCTION notify_invoice_overdue();

⸻

📌 Events Table (Core)

CREATE TABLE ai_events (
  id SERIAL PRIMARY KEY,
  type TEXT,
  payload JSONB,
  status TEXT DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT NOW()
);
