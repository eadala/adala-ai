🔐 4) نظام التشفير (API Keys Vault)

import crypto from "crypto";
const KEY = process.env.SECRET_KEY || "default_key";
export function encrypt(text: string) {
  const cipher = crypto.createCipheriv("aes-256-ecb", KEY, null);
  return cipher.update(text, "utf8", "hex") + cipher.final("hex");
}
export function decrypt(text: string) {
  const decipher = crypto.createDecipheriv("aes-256-ecb", KEY, null);
  return decipher.update(text, "hex", "utf8") + decipher.final("utf8");
}

🧠 5) AI Engine (RAG Core)

📚 Embedding Service

import OpenAI from "openai";
const openai = new OpenAI({ apiKey: process.env.OPENAI_KEY });
export async function embed(text: string) {
  const res = await openai.embeddings.create({
    model: "text-embedding-3-large",
    input: text,
  });
  return res.data[0].embedding;
}

⸻

🔍 AI Query (RAG Flow)

export async function askAI(query: string, context: string) {
  const openai = new OpenAI({ apiKey: process.env.OPENAI_KEY });
  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: "أنت مساعد قانوني ذكي." },
      { role: "user", content: context + "\n\nسؤال: " + query },
    ],
  });
  return response.choices[0].message.content;
}

📚 6) رفع الملفات + تحويلها (RAG Pipeline)

export async function processFile(fileText: string) {
  const chunks = fileText.match(/.{1,1000}/g) || [];
  const vectors = [];
  for (const chunk of chunks) {
    const vector = await embed(chunk);
    vectors.push({ chunk, vector });
  }
  return vectors;
}
