🧠 3. AGGREGATION ENGINE

يحول ledger إلى أرقام جاهزة

export async function getFinancialSummary(tenant_id) {
  const ledger = await db.ledger.findMany({
    where: { tenant_id }
  });
  const summary = ledger.reduce((acc, entry) => {
    acc.credit += entry.credit;
    acc.debit += entry.debit;
    return acc;
  }, { credit: 0, debit: 0 });
  return {
    totalRevenue: summary.credit,
    totalExpenses: summary.debit,
    net: summary.credit - summary.debit
  };
}

app.get("/api/finance/dashboard", async (req, res) => {
  const tenant_id = req.user.tenant_id;
  const summary = await getFinancialSummary(tenant_id);
  const invoices = await getInvoiceStats(tenant_id);
  const cashflow = await getCashFlow(tenant_id);
  res.json({
    summary,
    invoices,
    cashflow

});

📈 5. CASHFLOW ENGINE

export async function getCashFlow(tenant_id) {
  const transactions = await db.transaction.findMany({
    where: { tenant_id }
  });
  const daily = groupByDay(transactions);
  return daily.map(day => ({
    date: day.date,
    income: day.income,
    expenses: day.expenses,
    net: day.income - day.expenses
  }));
}

🧾 6. INVOICE ANALYTICS

export async function getInvoiceStats(tenant_id) {
  const invoices = await db.invoices.findMany({
    where: { tenant_id }
  });
  return {
    paid: count(invoices, "paid"),
    pending: count(invoices, "pending"),
    overdue: count(invoices, "overdue")
  };
}

export function generateInsights(data) {
  const insights = [];
  if (data.cashflow.trend === "down") {
    insights.push("⚠️ انخفاض في التدفق النقدي هذا الأسبوع");
  }
  if (data.invoices.overdue > 5) {
    insights.push("⚠️ عدد فواتير متأخرة مرتفع");
  }
  if (data.netProfit < 0) {
    insights.push("🔴 المصاريف أعلى من الإيرادات");
  }
  return insights;
}
