⚙️ 2) AI Gateway (القلب الحقيقي للنظام)

🔷 AI Entry Point

// ai-gateway/src/ai.service.ts
export class AIService {
  constructor(
    private router: AIRouter,
    private policy: PolicyEngine,
    private billing: BillingService
  ) {}
  async run(input: AIRequest) {
    const provider = this.policy.select(input);
    const result = await this.router.execute(provider, input.prompt);
    await this.billing.charge({
      tenantId: input.tenantId,
      provider,
      tokens: result.tokens,
    });
    return result.text;
  }
}

⸻

🔀 AI Router (Multi Provider System)

export class AIRouter {
  constructor(private providers: Record<string, AIProvider>) {}
  async execute(provider: string, prompt: string) {
    try {
      return await this.providers[provider].generate(prompt);
    } catch (e) {
      return this.fallback(prompt);
    }
  }
  async fallback(prompt: string) {
    return this.providers["gemini"].generate(prompt);
  }
}

⸻

🧠 Policy Engine (ذكاء القرار)

export class PolicyEngine {
  select(input: AIRequest) {
    if (input.type === "legal_contract") return "claude";
    if (input.cost_sensitive) return "gemini";
    if (input.fast_mode) return "gemini_flash";
    return "openai";
  }
}

🧮 Cost Engine

export class CostEngine {
  private rates = {
    openai: 0.002,
    claude: 0.003,
    gemini: 0.0005
  };
  calc(provider: string, tokens: number) {
    return tokens * this.rates[provider];
  }
}

💳 Wallet System

export class WalletService {
  async charge(tenantId: string, cost: number) {
    const tenant = await db.tenant.findUnique({ where: { id: tenantId } });
    if (tenant.balance < cost) {
      throw new Error("Insufficient balance");
    }
    await db.tenant.update({
      where: { id: tenantId },
      data: { balance: tenant.balance - cost }
    });
  }
}

export class LegalEngine {
  build(type: string, data: any) {
    const prompts = {
      contract: `اكتب عقد قانوني سعودي مطابق للأنظمة: ${data}`,
      lawsuit: `صغ لائحة دعوى قانونية: ${data}`,
      analysis: `حلل هذا المستند قانونياً: ${data}`
    };
    return prompts[type];
  }
}
