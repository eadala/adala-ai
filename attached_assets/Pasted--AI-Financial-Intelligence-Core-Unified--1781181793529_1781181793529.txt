⚙️ 2) المحرك الموحد (Core Engine)

import { PrismaClient } from "@prisma/client";
const prisma = new PrismaClient();
/**
 * 🔢 KPIs أساسية
 */
async function getKPIs() {
  const payments = await prisma.payment.findMany();
  const usage = await prisma.usageCost.findMany();
  const subs = await prisma.subscription.findMany();
  const revenue = payments.reduce((s, p) => s + p.amount, 0);
  const cost = usage.reduce((s, u) => s + u.cost, 0);
  return {
    revenue,
    cost,
    profit: revenue - cost,
    activeSubscriptions: subs.filter(s => s.status === "active").length
  };
}
/**
 * 📉 كشف الشذوذ
 */
async function detectAnomalies(usage) {
  const spikes = usage.filter(u => u.cost > 100);
  return {
    anomaliesCount: spikes.length,
    riskLevel: spikes.length > 0 ? "HIGH" : "LOW",
    flagged: spikes
  };
}
/**
 * 🔮 توقع بسيط (Forecasting)
 */
function forecastRevenue(revenue) {
  const growthFactor = 1.12; // تقديري
  return {
    nextMonth: revenue * growthFactor,
    confidence: "medium"
  };
}
/**
 * 🤖 توليد Insights ذكية
 */
function generateInsights(kpis, anomalies) {
  let insights = [];
  if (kpis.profit < 0) {
    insights.push("⚠️ النظام يعمل بخسارة حالياً");
  } else {
    insights.push("✅ النظام مربح ومستقر");
  }
  if (kpis.revenue > 10000) {
    insights.push("📈 نمو قوي في الإيرادات");
  }
  if (anomalies.anomaliesCount > 0) {
    insights.push("🚨 يوجد استهلاك غير طبيعي لبعض الخدمات");
  }
  if (kpis.activeSubscriptions < 10) {
    insights.push("📉 عدد الاشتراكات منخفض");
  }
  return insights;
}
/**
 * 🧠 المحرك الموحد النهائي
 */
export async function getUnifiedFinancialAI() {
  const kpis = await getKPIs();
  const usage = await prisma.usageCost.findMany();
  const anomalies = await detectAnomalies(usage);
  const forecast = forecastRevenue(kpis.revenue);
  const insights = generateInsights(kpis, anomalies);
  return {
    kpis,
    anomalies,
    forecast,
    aiInsights: insights
  };
}

🚀 3) API موحد واحد فقط (بدل 3 APIs)

import express from "express";
import { getUnifiedFinancialAI } from "../services/unified_finance_ai.js";
const router = express.Router();
/**
 * 🧠 Financial Intelligence API (Single Source of Truth)
 */
router.get("/financial-intelligence", async (req, res) => {
  const data = await getUnifiedFinancialAI();
  res.json({
    system: "AHDALA_AI_FINANCE_CORE",
    timestamp: new Date(),
    ...data
  });
});
export default router;
