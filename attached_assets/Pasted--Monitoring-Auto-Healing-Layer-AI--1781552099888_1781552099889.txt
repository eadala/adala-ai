4. 🚨 Anomaly Detection Engine

يكتشف:

* ارتفاع error rate
* بطء APIs
* فشل Webhooks
* زيادة retries
* عدم تطابق بيانات مالية

export function detectAnomalies(metrics) {
  const alerts = [];
  if (metrics.errorRate > 0.05) {
    alerts.push("HIGH_ERROR_RATE");
  }
  if (metrics.webhookFailures > 3) {
    alerts.push("STRIPE_WEBHOOK_FAILURE_SPIKE");
  }
  if (metrics.dbLatency > 500) {
    alerts.push("DATABASE_SLOWDOWN");
  }
  return alerts;
}

6. 🔁 Retry Engine (ذكي وليس عشوائي)

export async function retryWithBackoff(fn, retries = 3) {
  let delay = 1000;
  for (let i = 0; i < retries; i++) {
    try {
      return await fn();
    } catch (e) {
      await sleep(delay);
      delay *= 2;
    }
  }
  throw new Error("Max retries exceeded");
}
