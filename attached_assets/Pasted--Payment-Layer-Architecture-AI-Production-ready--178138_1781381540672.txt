🔌 4. Checkout.com Adapter

import axios from "axios";
export class CheckoutAdapter {
  private apiKey = process.env.CHECKOUT_SECRET_KEY;
  async createPayment(data: any) {
    const res = await axios.post(
      "https://api.checkout.com/payments",
      {
        amount: data.amount * 100,
        currency: data.currency,
        source: {
          type: "token",
          token: data.token,
        },
        reference: data.reference,
      },
      {
        headers: {
          Authorization: `Bearer ${this.apiKey}`,
          "Content-Type": "application/json",
        },
      }
    );
    return res.data;
  }
  async refund(paymentId: string) {
    return axios.post(
      `https://api.checkout.com/payments/${paymentId}/refunds`,
      {},
      {
        headers: {
          Authorization: `Bearer ${this.apiKey}`,
        },
      }
    );
  }
}

💳 5. Stripe Adapter (للمستقبل العالمي)

Stripe

import Stripe from "stripe";
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);
export class StripeAdapter {
  async createPayment(data: any) {
    return stripe.paymentIntents.create({
      amount: data.amount * 100,
      currency: data.currency,
      automatic_payment_methods: { enabled: true },
    });
  }
  async refund(paymentId: string) {
    return stripe.refunds.create({
      payment_intent: paymentId,
    });
  }
}

🧠 6. Payment Orchestrator (القلب الحقيقي)

import { CheckoutAdapter } from "./checkout";
import { StripeAdapter } from "./stripe";
class PaymentOrchestrator {
  private providers = {
    checkout: new CheckoutAdapter(),
    stripe: new StripeAdapter(),
  };
  async createPayment(provider: string, data: any) {
    return this.providers[provider].createPayment(data);
  }
  async refund(provider: string, paymentId: string) {
    return this.providers[provider].refund(paymentId);
  }
}
export const PaymentService = new PaymentOrchestrator();

⚖️ 7. منطق سوق المحاماة (أهم جزء)

export async function handleMarketplacePayment(tx) {
  const platformFee = tx.amount * 0.10;
  const lawyerAmount = tx.amount - platformFee;
  // تسجيل في ledger
  await Wallet.credit("platform", platformFee);
  await Wallet.credit(tx.lawyerId, lawyerAmount);
  // إنشاء payout
  await Payout.create({
    lawyerId: tx.lawyerId,
    transactionId: tx.id,
    platformFee,
    lawyerAmount,
    status: "pending",
  });
}

🔔 8. Webhooks (ضروري جداً)

app.post("/webhooks/payment", async (req, res) => {
  const event = req.body;
  if (event.type === "payment.succeeded") {
    await Transaction.updateStatus(event.data.id, "paid");
    if (event.data.type === "marketplace") {
      await handleMarketplacePayment(event.data);
    }
  }
  res.sendStatus(200);
});
