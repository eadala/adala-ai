# 1. جدول GitHub Settings

## `shared/schema.ts`

```ts
export const githubSettings = pgTable(
  "github_settings",
  {

    id: uuid("id")
      .defaultRandom()
      .primaryKey(),

    githubUsername: varchar(
      "github_username",
      { length: 255 }
    ).notNull(),

    repository: varchar(
      "repository",
      { length: 255 }
    ).notNull(),

    branch: varchar(
      "branch",
      { length: 100 }
    )
      .default("main")
      .notNull(),

    accessToken: text(
      "access_token"
    ).notNull(),

    autoPull: boolean(
      "auto_pull"
    ).default(false),

    autoBackup: boolean(
      "auto_backup"
    ).default(true),

    autoRestart: boolean(
      "auto_restart"
    ).default(true),

    createdAt: timestamp(
      "created_at"
    ).defaultNow()
  }
);
```

# 2. Service

## `server/services/githubService.ts`

```ts
import simpleGit from "simple-git";

const git = simpleGit();

export async function pullLatestChanges() {

  await git.pull(
    "origin",
    "main"
  );

  return {
    success: true
  };

}

export async function getCurrentBranch() {

  return await git.branch();

}

export async function getCommits() {

  return await git.log();

}

export async function pushChanges(
  message: string
) {

  await git.add(".");

  await git.commit(message);

  await git.push(
    "origin",
    "main"
  );

}
```

# 3. Routes

## `server/routes/github.ts`

```ts
import express from "express";

import {

pullLatestChanges,
getCurrentBranch,
getCommits

} from "../services/githubService";

const router = express.Router();

router.get(
"/status",
async (
req,
res
) => {

const branch =
await getCurrentBranch();

res.json(branch);

}
);

router.get(
"/commits",
async (
req,
res
) => {

const commits =
await getCommits();

res.json(commits);

}
);

router.post(
"/pull",
async (
req,
res
) => {

await pullLatestChanges();

res.json({
success: true
});

}
);

export default router;
```

# 7. واجهة React

## `client/src/pages/admin/GithubPage.tsx`

import {

Button

} from "@/components/ui/button";

        method: "POST"
      }
    );

  }

  return (

    <div className="space-y-6">

      <h1 className="text-3xl font-bold">

        GitHub Integration

      </h1>

      <Button
        onClick={pullUpdates}
