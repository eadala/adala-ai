2. أهم نقطة احترافية (RLS)

بدل ما تعتمد فقط على WHERE conditions…

نستخدم:

🔥 PostgreSQL Row Level Security

ALTER TABLE cases ENABLE ROW LEVEL SECURITY;
CREATE POLICY branch_isolation ON cases
USING (
  tenant_id = current_setting('app.tenant_id')::text
  AND branch_id = current_setting('app.branch_id')::text
);

✔ هذا يمنع أي تسريب بيانات حتى لو فيه bug في الكود

3. Context Middleware (قلب النظام)

export function setTenantContext(req, res, next) {
  const tenantId = req.headers["x-tenant-id"];
  const branchId = req.headers["x-branch-id"];
  const userId = req.headers["x-user-id"];
  if (!tenantId) return res.status(401).json({ error: "No tenant" });
  req.context = {
    tenantId,
    branchId,
    userId,
  };
  // تمريرها لـ PostgreSQL RLS
  req.dbQuery = async (query, params) => {
    await db.query(`SELECT set_config('app.tenant_id', $1, true)`, [tenantId]);
    await db.query(`SELECT set_config('app.branch_id', $1, true)`, [branchId]);
    return db.query(query, params);
  };
  next();
}

6. API احترافي لإنشاء فرع

app.post("/branches", async (req, res) => {
  const { tenantId } = req.context;
  const { name, location, managerUserId } = req.body;
  const existing = await req.dbQuery(
    `SELECT COUNT(*) FROM office_branches WHERE tenant_id = $1`,
    [tenantId]
  );
  const plan = await getTenantPlan(tenantId);
  if (plan.branches !== "unlimited" &&
      parseInt(existing.rows[0].count) >= plan.branches) {
    return res.status(403).json({
      error: "تم تجاوز حد الفروع في الباقة"
    });
  }
  const result = await req.dbQuery(
    `INSERT INTO office_branches (tenant_id, name, location, manager_user_id)
     VALUES ($1, $2, $3, $4)
     RETURNING *`,
    [tenantId, name, location, managerUserId]
  );
  res.json(result.rows[0]);
});

9. نقل القضايا بين الفروع (Enterprise Feature)

app.post("/cases/transfer", async (req, res) => {
  const { caseId, targetBranchId } = req.body;
  await req.dbQuery(
    `UPDATE cases
     SET branch_id = $1
     WHERE id = $2`,
    [targetBranchId, caseId]
  );
  res.json({ success: true });
});
