🧠 Adala SaaS Kernel (Core Architecture)

الهدف

تحقيق 4 ضمانات لا يمكن كسرها:

1. ❌ لا يمكن الوصول لأي بيانات بدون Tenant
2. ❌ لا يمكن تنفيذ Query بدون فلترة مكتب
3. ❌ لا يمكن تجاوز الصلاحيات حتى لو حدث bug
4. ❌ كل عملية يتم تسجيلها (Audit Trail)

🔴 1) Tenant Context Kernel (مصدر الحقيقة)

import { AsyncLocalStorage } from "node:async_hooks";
export const TenantContext = new AsyncLocalStorage();
export function runWithTenant(context, callback) {
  return TenantContext.run(context, callback);
}
export function getTenant() {
  const store = TenantContext.getStore();
  if (!store) throw new Error("NO_TENANT_CONTEXT");
  return store;
}

🔥 2) Kernel Middleware (إجبار السياق)

export function tenantKernel(req, res, next) {
  const userId = req.auth?.userId;
  const officeId = req.auth?.sessionClaims?.officeId;
  if (!userId || !officeId) {
    return res.status(401).json({ error: "Unauthorized Tenant Access" });
  }
  runWithTenant({ userId, officeId }, () => next());
}

🧱 3) Secure DB Layer (قلب النظام)

❌ ممنوع استخدام db مباشرة

✅ استخدام Kernel DB

export function createKernelDB(db) {
  return new Proxy(db, {
    get(target, tableName) {
      const table = target[tableName];
      return {
        findMany: async (builder = (q) => q) => {
          const tenant = getTenant();
          let query = db.select().from(table);
          query = query.where(eq(table.officeId, tenant.officeId));
          query = builder(query);
          return query;
        },
        findOne: async (id) => {
          const tenant = getTenant();
          return db
            .select()
            .from(table)
            .where(
              and(
                eq(table.id, id),
                eq(table.officeId, tenant.officeId)
              )
            );
        },
        insert: async (data) => {
          const tenant = getTenant();
          return db.insert(table).values({
            ...data,
            officeId: tenant.officeId, // 🔥 forced injection
          });
        },
      };
    },
  });
}
