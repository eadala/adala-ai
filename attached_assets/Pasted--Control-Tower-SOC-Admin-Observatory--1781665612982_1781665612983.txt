🔐 Middleware حماية

export function requireControlTower(req, res, next) {
  const user = req.user;
  if (
    user.role !== "CONTROL_TOWER_ADMIN" &&
    user.role !== "PLATFORM_OWNER"
  ) {
    return res.status(403).json({ error: "FORBIDDEN" });
  }
  next();
}

router.get("/stream", requireControlTower, (req, res) => {
  res.setHeader("Content-Type", "text/event-stream");
  const interval = setInterval(() => {
    res.write(`data: ${JSON.stringify(getMetrics())}\n\n`);
  }, 2000);
  req.on("close", () => clearInterval(interval));
});

📊 5) Control Tower UI (لوحة احترافية)

📁 client/pages/ControlTower.tsx

import { useEffect, useState } from "react";
export default function ControlTower() {
  const [data, setData] = useState<any>(null);
  useEffect(() => {
    const ws = new EventSource("/api/control-tower/stream");
    ws.onmessage = (e) => {
      setData(JSON.parse(e.data));
    };
    return () => ws.close();
  }, []);
  return (
    <div className="min-h-screen bg-white p-6 grid grid-cols-3 gap-4">
      <div className="card">
        <h2>System Health</h2>
        <p>{data?.systemHealth}</p>
      </div>
      <div className="card">
        <h2>Active Users</h2>
        <p>{data?.activeUsers}</p>
      </div>
      <div className="card text-red-600">
        <h2>Security Events</h2>
        <p>{data?.securityEvents}</p>
      </div>
      <div className="col-span-3 border p-4">
        <h2>Tenant Activity Live Feed</h2>
        <pre>{JSON.stringify(data?.tenantActivity, null, 2)}</pre>
      </div>
    </div>
  );
}
