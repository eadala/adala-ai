🚦 6. Middleware (أهم جزء)

middleware.ts

import { NextResponse } from "next/server";
export function middleware(req) {
  const role = req.cookies.get("role")?.value;
  const path = req.nextUrl.pathname;
  if (path.startsWith("/admin")) {
    if (role !== "platform_admin") {
      return NextResponse.redirect(new URL("/not-authorized", req.url));
    }
  }
  if (path.startsWith("/workspace")) {
    if (!["law_firm_owner", "lawyer"].includes(role || "")) {
      return NextResponse.redirect(new URL("/not-authorized", req.url));
    }
  }
  if (path.startsWith("/client")) {
    if (role !== "client") {
      return NextResponse.redirect(new URL("/not-authorized", req.url));
    }
  }
  return NextResponse.next();
}

🔀 7. Smart Redirect بعد تسجيل الدخول

lib/redirect.ts

export function redirectByRole(role: string) {
  switch (role) {
    case "platform_admin":
      return "/admin";
    case "law_firm_owner":
    case "lawyer":
      return "/workspace";
    case "client":
      return "/client";
    default:
      return "/login";
  }
}

🔐 9. Login Page

app/login/page.tsx

"use client";
import { redirectByRole } from "@/lib/redirect";
export default function Login() {
  function login(role: string) {
    document.cookie = `role=${role}`;
    window.location.href = redirectByRole(role);
  }
  return (
    <div className="p-10 space-y-4">
      <h1 className="text-2xl">Login</h1>
      <button onClick={() => login("platform_admin")}>
        Admin Login
      </button>
      <button onClick={() => login("law_firm_owner")}>
        Lawyer Login
      </button>
      <button onClick={() => login("client")}>
        Client Login
      </button>
    </div>
  );
}
