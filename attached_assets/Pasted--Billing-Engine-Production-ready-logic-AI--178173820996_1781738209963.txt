🏢 4) Branch Pricing Engine (IMPORTANT)

// billing/branches.ts
export function calculateBranchCost(
  includedBranches: number,
  totalBranches: number,
  basePrice: number
) {
  if (totalBranches <= includedBranches) return 0;
  const extra = totalBranches - includedBranches;
  let cost = 0;
  for (let i = 1; i <= extra; i++) {
    if (i <= 3) cost += 200;      // أول 3 فروع إضافية
    else if (i <= 7) cost += 180;
    else cost += 150;
  }
  return cost;
}

🤖 5) AI Credits Engine

// billing/ai.ts
export function calculateAICost(
  usedCredits: number,
  includedCredits: number
) {
  if (usedCredits <= includedCredits) return 0;
  const extra = usedCredits - includedCredits;
  // 1 credit unit cost
  const costPer1000 = 0.02; // ريال
  return (extra / 1000) * costPer1000;
}

🧮 6) Main Billing Engine

// billing/calculateBill.ts
import { PLANS } from "./plans";
import { ADDON_PRICES, AddOns } from "./addons";
import { calculateBranchCost } from "./branches";
import { calculateAICost } from "./ai";
interface BillingInput {
  planId: string;
  users: number;
  storageGB: number;
  aiUsedCredits: number;
  branches: number;
  addons: AddOns;
}
export function calculateBill(input: BillingInput) {
  const plan = PLANS[input.planId];
  if (!plan) throw new Error("Invalid plan");
  let total = plan.basePrice;
  // 🔗 Add-ons
  if (input.addons.whatsapp) total += ADDON_PRICES.whatsapp;
  if (input.addons.telegram) total += ADDON_PRICES.telegram;
  if (input.addons.accounting) total += ADDON_PRICES.accounting;
  if (input.addons.hr) total += ADDON_PRICES.hr;
  // 🏢 Branches
  total += calculateBranchCost(
    plan.includedBranches,
    input.branches,
    plan.basePrice
  );
  // 🤖 AI usage
  total += calculateAICost(
    input.aiUsedCredits,
    plan.includedAIcredits
  );
  return {
    plan: plan.name,
    total: Math.round(total),
    breakdown: {
      base: plan.basePrice,
      addons: input.addons,
      branches: input.branches,
      aiUsedCredits: input.aiUsedCredits,
    },
  };
}
