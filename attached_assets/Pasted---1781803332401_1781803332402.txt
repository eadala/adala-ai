تمام — هذا هو كود احترافي جاهز لتنفيذ “مزامنة الإنتاج → الديمو (الشمال/الجنوب)” بشكل آمن وقابل للتوسّع داخل أي Node.js/TypeScript backend.

الفكرة:
Production = مصدر الحقيقة
Demo Tenants = Mirror Read-Only Replica (مع بيانات اختبار فقط)

🧠 3. جلب إعدادات الإنتاج (Source of Truth)

export class ProductionSource {
  static async fetchConfig(db: any) {
    const features = await db.features.findMany();
    const plans = await db.plans.findMany();
    const permissions = await db.permissions.findMany();
    return {
      features,
      plans,
      permissions,
      version: "prod-latest",
    };
  }
}

🔄 4. محرك المزامنة الأساسي

export class DemoSyncEngine {
  constructor(
    private prodDb: any,
    private demoDb: any
  ) {}
  async syncAll() {
    const prodData = await ProductionSource.fetchConfig(this.prodDb);
    for (const tenant of DEMO_TENANTS) {
      await this.syncTenant(tenant, prodData);
    }
    return {
      status: "SYNC_COMPLETE",
      syncedTenants: DEMO_TENANTS.length,
    };
  }
  async syncTenant(tenantId: string, prodData: any) {
    console.log(`🔄 Syncing tenant: ${tenantId}`);
    // 1. Feature flags sync
    await this.demoDb.featureFlags.upsertMany({
      where: { tenantId },
      create: prodData.features.map((f: any) => ({
        ...f,
        tenantId,
        isDemo: true,
      })),
      update: prodData.features,
    });
    // 2. Plans sync
    await this.demoDb.plans.deleteMany({ where: { tenantId } });
    await this.demoDb.plans.createMany({
      data: prodData.plans.map((p: any) => ({
        ...p,
        tenantId,
        isDemo: true,
      })),
    });
    // 3. Permissions sync
    await this.demoDb.permissions.deleteMany({ where: { tenantId } });
    await this.demoDb.permissions.createMany({
      data: prodData.permissions.map((p: any) => ({
        ...p,
        tenantId,
        isDemo: true,
      })),
    });
    console.log(`✅ Synced: ${tenantId}`);
  }
}

🚨 5. API Endpoint لتشغيل المزامنة

import express from "express";
const router = express.Router();
router.post("/admin/sync-demo", async (req, res) => {
  try {
    const engine = new DemoSyncEngine(
      req.app.locals.prodDb,
      req.app.locals.demoDb
    );
    const result = await engine.syncAll();
    res.json({
      success: true,
      result,
    });
  } catch (err: any) {
    res.status(500).json({
      success: false,
      error: err.message,
    });
  }
});
export default router;

export function requireAdmin(req: any, res: any, next: any) {
  if (req.user.role !== "admin") {
    return res.status(403).json({ error: "Forbidden" });
  }
  next();
}

🧠 8. حماية من اختلاف المستقبل (Drift Guard)

export class DriftGuard {
  static compare(prod: any, demo: any) {
    const issues = [];
    if (JSON.stringify(prod.features) !== JSON.stringify(demo.features)) {
      issues.push("FEATURE_DRIFT");
    }
    if (JSON.stringify(prod.plans) !== JSON.stringify(demo.plans)) {
      issues.push("PLAN_DRIFT");
    }
    return issues;
  }
}

📊 النتيجة

بعد هذا الكود:

✔️ أي تغيير في الإنتاج

→ يتم نسخه للديمو

✔️ مكتب الشمال والجنوب

→ يصبحان نسخة من الإنتاج

✔️ لا يوجد اختلاف بعد الآن
