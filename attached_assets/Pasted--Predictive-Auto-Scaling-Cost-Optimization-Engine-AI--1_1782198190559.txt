📊 4. System Metrics Collector

async function getMetrics() {
  const cpu = await queryProm("node_cpu_usage_percent");
  const mem = await queryProm("node_memory_usage_percent");
  const latency = await queryProm("http_request_duration_ms_p95");
  const errors = await queryProm("rate(http_requests_total{status=~'5..'}[5m])");
  return { cpu, mem, latency, errors };
}
async function queryProm(query: string) {
  const res = await axios.get(`${process.env.PROM_URL}/api/v1/query`, {
    params: { query },
  });
  return parseFloat(res.data.data.result[0]?.value[1] || "0");
}

🔮 5. Predict Failure Engine (Core Logic)

function predictFailure(metrics: any) {
  let riskScore = 0;
  if (metrics.cpu > 80) riskScore += 30;
  if (metrics.mem > 85) riskScore += 30;
  if (metrics.latency > 1000) riskScore += 20;
  if (metrics.errors > 0.05) riskScore += 20;
  return {
    riskScore,
    status:
      riskScore > 70
        ? "CRITICAL"
        : riskScore > 40
        ? "WARNING"
        : "STABLE",
  };
}

⚙️ 6. Auto Scaling Decision Engine

async function scaleSystem(metrics: any) {
  if (metrics.cpu > 75 || metrics.latency > 800) {
    await axios.post(process.env.SCALE_API!, {
      action: "scale_up",
    });
    return "scaled_up";
  }
  if (metrics.cpu < 30 && metrics.latency < 200) {
    await axios.post(process.env.SCALE_API!, {
      action: "scale_down",
    });
    return "scaled_down";
  }
  return "no_action";
}

💰 7. Cost Optimization Engine (مهم جداً)

function costOptimization(metrics: any) {
  let savings = [];
  if (metrics.cpu < 20) {
    savings.push("reduce_workers");
  }
  if (metrics.mem < 25) {
    savings.push("downsize_container");
  }
  if (metrics.idle_requests > 0.6) {
    savings.push("enable_sleep_mode");
  }
  return savings;
}
