🚨 4. Receive Alerts from Alertmanager

app.post("/alert", async (req, res) => {
  const alert = req.body;
  if (systemState.healing) {
    return res.json({ ignored: true });
  }
  if (alert.labels?.alertname === "APIDown") {
    return handleApiDown();
  }
  if (alert.labels?.severity === "critical") {
    return handleCritical(alert);
  }
  res.json({ received: true });
});

🔧 5. Intelligent Decision Logic

🔴 API DOWN

async function handleApiDown() {
  systemState.healing = true;
  await axios.post(
    "http://api:4000/internal/heal",
    {},
    {
      headers: {
        Authorization: `Bearer ${process.env.HEAL_SECRET}`,
      },
    }
  );
  return { action: "healing_triggered" };
}

⸻

🟠 CRITICAL ALERT (smart decision)

async function handleCritical(alert: any) {
  const memory = await getSystemMemory();
  if (memory > 90) {
    await restartWorkers();
    return { action: "workers_restarted" };
  }
  if (alert.annotations?.service === "db") {
    await scaleDbCheck();
    return { action: "db_check_triggered" };
  }
  return { action: "no_action_safe_mode" };
}

📊 6. System Health Snapshot Engine

async function getSystemMemory() {
  const res = await axios.get("http://prometheus:9090/api/v1/query", {
    params: {
      query: "node_memory_usage_percent",
    },
  });
  return parseFloat(res.data.data.result[0]?.value[1] || "0");
}

async function checkVersionSafety() {
