📊 2) API: لوحة الفوترة الرئيسية

import { PLANS } from "../plans/plans";
export async function getBillingOverview(user: any) {
  const plan = PLANS[user.plan];
  const usage = user.usage || {};
  return {
    plan: {
      name: user.plan,
      price: plan.price
    },
    limits: plan.limits,
    usage: {
      aiCallsToday: usage.aiCallsToday || 0,
      storageUsed: usage.storageUsed || 0,
      usersUsed: usage.usersUsed || 1,
      casesUsed: usage.casesUsed || 0
    },
    nextBillingDate: user.nextBillingDate,
    status: user.subscriptionStatus
  };
}

import { stripe } from "../stripe/stripe_client";
export async function getStripeSubscription(subscriptionId: string) {
  const subscription = await stripe.subscriptions.retrieve(subscriptionId);
  return {
    id: subscription.id,
    status: subscription.status,
    current_period_start: subscription.current_period_start,
    current_period_end: subscription.current_period_end,
    cancel_at_period_end: subscription.cancel_at_period_end
  };
}

📄 4) الفواتير (Invoices System)

import { stripe } from "../stripe/stripe_client";
export async function getInvoices(customerId: string) {
  const invoices = await stripe.invoices.list({
    customer: customerId,
    limit: 20
  });
  return invoices.data.map(inv => ({
    id: inv.id,
    amount: inv.amount_paid / 100,
    currency: inv.currency,
    status: inv.status,
    date: inv.created,
    pdf: inv.invoice_pdf
  }));
}

📈 5) الإيرادات (Admin Analytics)

import { stripe } from "../stripe/stripe_client";
export async function getRevenueAnalytics() {
  const charges = await stripe.charges.list({
    limit: 100
  });
  const totalRevenue = charges.data.reduce((sum, c) => {
    return sum + (c.amount / 100);
  }, 0);
  return {
    totalRevenue,
    transactions: charges.data.length
  };
}

⚠️ 6) نظام التنبيهات المالية (Billing Alerts)

export function checkBillingAlerts(user: any) {
  const alerts: string[] = [];
  if (user.subscriptionStatus === "past_due") {
    alerts.push("⚠️ يوجد دفعة متأخرة");
  }
  if (user.plan === "FREE" && user.usage.aiCallsToday > 5) {
    alerts.push("⚠️ تجاوزت حد الذكاء الاصطناعي");
  }
  if (user.paymentFailedCount > 0) {
    alerts.push("❌ فشل في عملية دفع سابقة");
  }
  return alerts;
}

🔁 7) الترقية داخل النظام (Upgrade Button)

import { stripe } from "../stripe/stripe_client";
import { STRIPE_PLANS } from "../stripe/stripe_plans_map";
export async function upgradePlan(userId: string, plan: string) {
  const priceId = STRIPE_PLANS[plan]?.priceId;
  if (!priceId) {
    throw new Error("Invalid plan");
  }
  const session = await stripe.checkout.sessions.create({
    mode: "subscription",
    line_items: [{ price: priceId, quantity: 1 }],
    success_url: `${process.env.APP_URL}/billing/success`,
    cancel_url: `${process.env.APP_URL}/billing/cancel`,
    metadata: {
      userId,
      plan
    }
  });
  return session.url;
}
