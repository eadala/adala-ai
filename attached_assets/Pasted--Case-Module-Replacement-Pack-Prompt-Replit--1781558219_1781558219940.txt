🧩 4. CASE REPOSITORY (العزل الإجباري)

export class CaseRepository {
  constructor(private db: any, private tenantId: string) {}
  async findAll() {
    return this.db.cases.findMany({
      where: { tenant_id: this.tenantId }
    });
  }
  async findById(id: string) {
    return this.db.cases.findFirst({
      where: {
        id,
        tenant_id: this.tenantId
      }
    });
  }
  async create(data: Partial<any>) {
    return this.db.cases.create({
      data: {
        ...data,
        tenant_id: this.tenantId
      }
    });
  }
  async update(id: string, data: any) {
    return this.db.cases.update({
      where: {
        id,
        tenant_id: this.tenantId
      },
      data
    });
  }
  async delete(id: string) {
    return this.db.cases.delete({
      where: {
        id,
        tenant_id: this.tenantId
      }
    });
  }
}

⚙️ 5. CASE SERVICE (القلب)

import { CaseRepository } from "./case.repository";
import { CaseEvents } from "./case.events";
export class CaseService {
  constructor(private repo: CaseRepository) {}
  async getCases() {
    return this.repo.findAll();
  }
  async getCase(id: string) {
    return this.repo.findById(id);
  }
  async createCase(data: any) {
    const newCase = await this.repo.create({
      ...data,
      status: "active"
    });
    CaseEvents.emit("CASE_CREATED", newCase);
    return newCase;
  }
  async updateCase(id: string, data: any) {
    const updated = await this.repo.update(id, data);
    CaseEvents.emit("CASE_UPDATED", updated);
    return updated;
  }
  async deleteCase(id: string) {
    const result = await this.repo.delete(id);
    CaseEvents.emit("CASE_DELETED", { id });
    return result;
  }
}
