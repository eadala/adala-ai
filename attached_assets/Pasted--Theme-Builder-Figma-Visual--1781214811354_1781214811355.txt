⚛️ 4. Theme Context (Dynamic Engine)

import { createContext, useContext, useState, useEffect } from "react";
const ThemeContext = createContext();
export function ThemeProvider({ children }) {
  const [tokens, setTokens] = useState(null);
  useEffect(() => {
    fetch("/api/theme")
      .then(res => res.json())
      .then(data => {
        setTokens(data);
        applyTokens(data);
      });
  }, []);
  const applyTokens = (t) => {
    const root = document.documentElement;
    // Colors
    root.style.setProperty("--primary", t.colors.primary);
    root.style.setProperty("--secondary", t.colors.secondary);
    root.style.setProperty("--accent", t.colors.accent);
    root.style.setProperty("--bg", t.colors.background);
    root.style.setProperty("--text", t.colors.text);
    // Typography
    root.style.setProperty("--font", t.typography.fontFamily);
    root.style.setProperty("--heading-font", t.typography.headingFont);
    // Radius
    root.style.setProperty("--radius", t.radius.card);
  };
  const updateTokens = (newTokens) => {
    setTokens(newTokens);
    applyTokens(newTokens);
  };
  return (
    <ThemeContext.Provider value={{ tokens, updateTokens }}>
      {children}
    </ThemeContext.Provider>
  );
}
export const useTheme = () => useContext(ThemeContext);

🎛️ Right Panel (Inspector)

export function Inspector({ selected, tokens, setTokens }) {
  const updateColor = (key, value) => {
    setTokens({
      ...tokens,
      colors: {
        ...tokens.colors,
        [key]: value
      }
    });
  };
  return (
    <div className="panel">
      <h3>⚙️ Properties</h3>
      <label>Primary Color</label>
      <input
        type="color"
        value={tokens.colors.primary}
        onChange={(e) => updateColor("primary", e.target.value)}
      />
      <label>Accent</label>
      <input
        type="color"
        value={tokens.colors.accent}
        onChange={(e) => updateColor("accent", e.target.value)}
      />
    </div>
  );
}
