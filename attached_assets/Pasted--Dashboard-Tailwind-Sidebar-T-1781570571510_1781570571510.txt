⚖️ 6) Case Card (محسن نهائي)

import Card from "./Card";
export default function CaseCard({ caseItem }) {
  return (
    <Card>
      <div className="text-lg font-semibold">
        قضية رقم #{caseItem.number}
      </div>
      <span className={`
        text-xs px-3 py-1 rounded-full inline-block mt-2
        ${caseItem.statusType === "warning" ? "bg-yellow-100 text-yellow-800" : ""}
        ${caseItem.statusType === "success" ? "bg-green-100 text-green-800" : ""}
      `}>
        {caseItem.status}
      </span>
      <div className="text-sm text-gray-500 mt-3 leading-relaxed">
        المحكمة: {caseItem.court}
        <br />
        آخر تحديث: {caseItem.updatedAt}
      </div>
      <button className="
        mt-4 bg-primary text-white
        px-4 py-2 rounded-md text-sm
        hover:opacity-90
      ">
        عرض التفاصيل
      </button>
    </Card>
  );
}
