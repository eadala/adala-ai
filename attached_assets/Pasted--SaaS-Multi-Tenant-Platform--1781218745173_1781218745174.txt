⚙️ 3. Middleware العزل (Core SaaS Layer)

function tenantMiddleware(req, res, next) {
  // استخراج المكتب من التوكن أو الدومين
  const firm_id = req.headers["x-firm-id"];
  if (!firm_id) {
    return res.status(400).json({ error: "Missing firm context" });
  }
  req.firm_id = firm_id;
  next();
}

📊 7. القوائم المالية (Multi-Tenant Ready)

app.post("/api/financial/report", tenantMiddleware, async (req, res) => {
  const { startDate, endDate } = req.body;
  const invoices = await db.invoices.findMany({
    where: {
      firm_id: req.firm_id,
      date: { gte: startDate, lte: endDate }
    }
  });
  const payments = await db.payments.findMany({
    where: { firm_id: req.firm_id }
  });
  const revenue = payments.reduce((s, p) => s + p.amount, 0);
  const receivables = invoices
    .filter(i => !i.is_paid)
    .reduce((s, i) => s + i.total, 0);
  const report = {
    firm_id: req.firm_id,
    income_statement: {
      revenue,
      expenses: 0,
      net_profit: revenue
    },
    balance_sheet: {
      assets: {
        cash: revenue,
        receivables
      },
      equity: revenue + receivables
    }
  };
  res.json(report);
});

🧠 10. AI Insights (لكل مكتب بشكل منفصل)

function analyzeFirm(report) {
  const insights = [];
  if (report.income_statement.revenue > 500000) {
    insights.push("📈 أداء مالي قوي لهذا المكتب");
  }
  if (report.balance_sheet.assets.receivables > report.income_statement.revenue * 0.3) {
    insights.push("⚠️ تأخر في تحصيل المستحقات");
  }
  return insights;
}
