  res.json({ status: "Legal SaaS Running 🚀" });

app.listen(3000, () => console.log("Server running on port 3000"));

💰 6) إنشاء الدفع (Checkout Session)

import express from "express";
import { stripe } from "../config/stripe.js";
const router = express.Router();
/**
 * Plans
 */
const PLANS = {
  basic: 149,
  pro: 699,
  enterprise: 9999
};
router.post("/checkout", async (req, res) => {
  const { plan, email } = req.body;
  const session = await stripe.checkout.sessions.create({
    payment_method_types: ["card"],
    mode: "payment",
    customer_email: email,
    metadata: {
      plan
    },
    line_items: [
      {
        price_data: {
          currency: "sar",
          product_data: {
            name: `عدالة AI - ${plan}`
          },
          unit_amount: PLANS[plan] * 100
        },
        quantity: 1
      }
    ],
    success_url: `${process.env.BASE_URL}/success`,
    cancel_url: `${process.env.BASE_URL}/cancel`
  });
  res.json({ url: session.url });
});
export default router;

🧠 7) أهم جزء: Webhook (التفعيل التلقائي)

import express from "express";
import { stripe } from "../config/stripe.js";
import { createTenantFromPayment } from "../services/tenant.service.js";
const router = express.Router();
router.post("/", async (req, res) => {
  const sig = req.headers["stripe-signature"];
  let event;
  try {
    event = stripe.webhooks.constructEvent(
      req.body,
      sig,
      process.env.STRIPE_WEBHOOK_SECRET
    );
  } catch (err) {
    return res.status(400).send(err.message);
  }
  /**
   * نجاح الدفع
   */
  if (event.type === "checkout.session.completed") {
    const session = event.data.object;
    await createTenantFromPayment(session);
  }
  res.json({ received: true });
});
export default router;

🏢 8) إنشاء مكتب تلقائي (Core SaaS Logic)

import { PrismaClient } from "@prisma/client";
import { generateApiKey } from "../utils/apiKey.js";
const prisma = new PrismaClient();
export async function createTenantFromPayment(session) {
  const plan = session.metadata.plan;
  const email = session.customer_email;
  // 1) إنشاء مكتب
  const tenant = await prisma.tenant.create({
    data: {
      name: `مكتب ${email}`,
      email,
      plan,
      status: "active"
    }
  });
  // 2) اشتراك
  await prisma.subscription.create({
    data: {
      tenantId: tenant.id,
      plan,
      status: "active"
    }
  });
  // 3) Entitlements (صلاحيات حسب الباقة)
  const limits =
    plan === "enterprise"
      ? { ai: 10000, cases: 9999 }
      : plan === "pro"
      ? { ai: 1000, cases: 500 }
      : { ai: 100, cases: 50 };
  await prisma.entitlement.createMany({
    data: [
      { tenantId: tenant.id, key: "AI", limit: limits.ai },
      { tenantId: tenant.id, key: "CASES", limit: limits.cases }
    ]
  });
  // 4) API Key
  const apiKey = generateApiKey();
  console.log("TENANT CREATED:", tenant.id, apiKey);
  return tenant;
}

import { PrismaClient } from "@prisma/client";
const prisma = new PrismaClient();
export async function checkAccess(tenantId) {
  const sub = await prisma.subscription.findFirst({
    where: { tenantId, status: "active" }
  });
  if (!sub) {
    throw new Error("Subscription expired");
  }
  return true;
}
