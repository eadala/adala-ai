⚙️ الكود

import { db } from "../db";
import { decryptBuffer } from "../backup/encrypt";
import { downloadFile } from "../backup/storage.s3";
export async function restoreTenant(req, res) {
  const requester = req.user;
  const tenantId = req.params.tenantId;
  // 🔐 RBAC CHECK
  if (!requester || requester.role !== "owner") {
    return res.status(403).json({ error: "Unauthorized restore operation" });
  }
  // 📦 تحميل النسخة الاحتياطية
  const backupKey = `tenants/${tenantId}/latest-snapshot.json.enc`;
  const encrypted = await downloadFile(backupKey);
  // 🔓 فك التشفير
  const decrypted = decryptBuffer(encrypted);
  const snapshot = JSON.parse(decrypted.toString());
  // 🧹 تنظيف البيانات القديمة (اختياري حسب السياسة)
  await db.delete(cases).where(eq(cases.tenantId, tenantId));
  await db.delete(clients).where(eq(clients.tenantId, tenantId));
  // ♻️ إعادة الإدخال
  for (const c of snapshot.cases) {
    await db.insert(cases).values({
      ...c,
      tenantId
    });
  }
  for (const cl of snapshot.clients) {
    await db.insert(clients).values({
      ...cl,
      tenantId
    });
  }
  return res.json({
    success: true,
    restored: {
      cases: snapshot.cases.length,
      clients: snapshot.clients.length
    }
  });
}

🔓 3️⃣ فك التشفير

import crypto from "crypto";
export function decryptBuffer(buffer: Buffer): Buffer {
  const key = crypto.createHash("sha256").update(process.env.BACKUP_ENCRYPTION_KEY!).digest();
  const iv = buffer.subarray(0, 16);
  const data = buffer.subarray(16);
  const decipher = crypto.createDecipheriv("aes-256-cbc", key, iv);
  return Buffer.concat([
    decipher.update(data),
    decipher.final()
  ]);
}

🧩 4️⃣ Restore Single Case (مهم جداً)

export async function restoreCase(caseId: string, tenantId: string) {
  const backup = await db.query.caseBackups.findFirst({
    where: (c, { eq }) => eq(c.caseId, caseId)
  });
  if (!backup) throw new Error("Backup not found");
  await db.insert(cases).values({
    ...backup.data,
    tenantId
  });
}

⚠️ 7️⃣ سيناريوهات الأمان

🟥 إذا حاول تعديل backup:

🟥 إذا backup تالف:
