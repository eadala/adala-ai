2️⃣ tenant-context.ts

export class TenantContext {
  officeId: string;
  constructor(officeId: string) {
    if (!officeId) throw new Error("Missing tenant context");
    this.officeId = officeId;
  }
  validateAccess(requestOfficeId: string) {
    if (requestOfficeId !== this.officeId) {
      throw new Error("TENANT_ISOLATION_VIOLATION");
    }
  }
}

3️⃣ context-builder.ts

import { db } from "../db";
export async function buildAIContext(officeId: string) {
  const cases = await db.cases.findMany({ where: { office_id: officeId } });
  const clients = await db.clients.findMany({ where: { office_id: officeId } });
  const invoices = await db.invoices.findMany({ where: { office_id: officeId } });
  return {
    cases,
    clients,
    invoices,
  };
}

4️⃣ isolated-memory.ts

const memoryStore: Record<string, any[]> = {};
export class IsolatedMemory {
  static get(officeId: string) {
    return memoryStore[officeId] || [];
  }
  static add(officeId: string, entry: any) {
    if (!memoryStore[officeId]) memoryStore[officeId] = [];
    memoryStore[officeId].push(entry);
  }
}

5️⃣ ai-tenant-guard.ts (أهم ملف أمني)

import { TenantContext } from "../tenant-context";
export function aiTenantGuard(req: any, res: any, next: any) {
  const officeId = req.user?.officeId;
  if (!officeId) {
    return res.status(403).json({ error: "Missing tenant context" });
  }
  req.aiTenant = new TenantContext(officeId);
  next();
}

8️⃣ router.ts

import { legalAgent } from "./agents/legal-agent";
import { devCommander } from "./agents/dev-commander";
export async function aiRouter(type: string, officeId: string, message: string) {
  switch (type) {
    case "legal":
      return legalAgent(officeId, message);
    case "dev":
      return devCommander(message);
    default:
      return "Agent not found";
  }
}

9️⃣ orchestrator.ts (العقل الرئيسي)

import { aiRouter } from "./router";
import { IsolatedMemory } from "./memory/isolated-memory";
export async function aiOrchestrator(req: any) {
  const { officeId, message } = req.body;
  const role = req.user.role;
  if (!officeId) throw new Error("Missing tenant");
  // حفظ السياق
  IsolatedMemory.add(officeId, {
    message,
    time: new Date(),
  });
  // اختيار الوكيل حسب نوع الرسالة (بسيط كبداية)
  const agentType = message.includes("case") ? "legal" : "dev";
  const response = await aiRouter(agentType, officeId, message);
  return {
    officeId,
    agent: agentType,
    response,
  };
}

🔟 index.ts (API endpoint)

import express from "express";
import { aiOrchestrator } from "./orchestrator";
import { aiTenantGuard } from "./middleware/ai-tenant-guard";
const router = express.Router();
router.post("/ai/command", aiTenantGuard, async (req, res) => {
  try {
    const result = await aiOrchestrator(req);
    res.json(result);
  } catch (err: any) {
    res.status(500).json({ error: err.message });
  }
});
export default router;
