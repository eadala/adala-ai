⚙️ الكود الأساسي

import { db } from "../db";
type TenantResult = {
  tenantId: string;
  role: string;
  source: "office_members" | "registry";
};
export async function resolveTenant(userId: string): Promise<TenantResult> {
  // 🟢 1. المصدر الأساسي (الموثوق)
  const member = await db.query.office_members.findFirst({
    where: (t, { eq }) => eq(t.userId, userId)
  });
  if (member) {
    return {
      tenantId: member.officeId,
      role: member.role,
      source: "office_members"
    };
  }
  // 🟡 2. مصدر ثانوي (فقط عند onboarding)
  const registry = await db.query.office_registry.findFirst({
    where: (t, { eq }) => eq(t.userId, userId)
  });
  if (registry) {
    // إنشاء ربط رسمي (مرة واحدة فقط)
    const officeMember = await db.insert(office_members).values({
      userId,
      officeId: registry.officeId,
      role: registry.role ?? "member"
    }).returning();
    return {
      tenantId: officeMember[0].officeId,
      role: officeMember[0].role,
      source: "registry"
    };
  }
  // 🔴 لا fallback إطلاقاً
  throw new Error("TENANT_NOT_RESOLVED");
}

⚡ 3️⃣ Middleware إجباري (Hard Gate)

import { resolveTenant } from "../core/tenant/tenant-resolver";
export async function requireTenant(req, res, next) {
  try {
    const userId = req.user.id;
    const tenant = await resolveTenant(userId);
    req.tenantId = tenant.tenantId;
    req.role = tenant.role;
    return next();
  } catch (err) {
    return res.status(403).json({
      error: "NO_TENANT_ACCESS"
    });
  }
}

🧠 4️⃣ Cache Layer (مهم للأداء)

import Redis from "ioredis";
const redis = new Redis(process.env.REDIS_URL);
export async function getCachedTenant(userId: string) {
  const cached = await redis.get(`tenant:${userId}`);
  if (cached) return JSON.parse(cached);
  return null;
}
export async function setCachedTenant(userId: string, data: any) {
  await redis.set(`tenant:${userId}`, JSON.stringify(data), "EX", 3600);
}

🚀 5️⃣ تحسين Middleware (نسخة إنتاجية)

export async function requireTenant(req, res, next) {
  try {
    const userId = req.user.id;
    const cached = await getCachedTenant(userId);
    let tenant;
    if (cached) {
      tenant = cached;
    } else {
      tenant = await resolveTenant(userId);
      await setCachedTenant(userId, tenant);
    }
    req.tenantId = tenant.tenantId;
    req.role = tenant.role;
    return next();
  } catch (err) {
    return res.status(403).json({
      error: "TENANT_NOT_RESOLVED"
    });
  }
}
