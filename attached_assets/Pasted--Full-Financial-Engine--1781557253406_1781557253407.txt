⚙️ 3. FINANCIAL ENGINE CORE

🧠 المعالج الرئيسي

export async function processTransaction(tx) {
  // 1. idempotency check
  const exists = await db.transaction.findUnique({
    where: { stripe_event_id: tx.stripe_event_id }
  });
  if (exists) return;
  // 2. save transaction
  const transaction = await db.transaction.create(tx);
  // 3. calculate revenue split
  const split = calculateRevenue(tx.amount);
  // 4. update ledger
  await createLedgerEntry({
    tenant_id: tx.tenant_id,
    transaction_id: transaction.id,
    credit: split.net,
    debit: 0
  });
  // 5. update balances
  await updateTenantBalance(tx.tenant_id);
  return transaction;
}

💰 4. REVENUE CALCULATION

export function calculateRevenue(amount: number) {
  const platform_fee = amount * 0.10;
  const stripe_fee = amount * 0.029 + 1.0;
  return {
    gross: amount,
    platform_fee,
    stripe_fee,
    net: amount - platform_fee - stripe_fee
  };
}

📒 6. LEDGER ENGINE (مصدر الحقيقة)

export async function createLedgerEntry(data) {
  const lastBalance = await getLastBalance(data.tenant_id);
  const entry = await db.ledger.create({
    ...data,
    debit: data.debit || 0,
    credit: data.credit || 0,
    balance_after:
      lastBalance + (data.credit - data.debit)
  });
  return entry;
}

🧾 7. TENANT BALANCE SYSTEM

export async function updateTenantBalance(tenant_id) {
  const balance = await db.ledger.aggregate({
    where: { tenant_id },
    _sum: {
      credit: true,
      debit: true
    }
  });
  return (balance._sum.credit || 0) -
         (balance._sum.debit || 0);
}
