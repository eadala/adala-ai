تمام—سنحوّل النظام من “قائمة بسيطة” إلى نظام تتبع دخول احترافي (Login Tracking System) بمستوى SaaS فعلي:

* تسجيل دخول + OTP (Mock حاليًا)
* حفظ سجل الدخول في قاعدة بيانات (LocalStorage الآن + جاهز للـ DB)
* تتبع IP + User Agent (جهاز المستخدم)
* ربط كل دخول بمكتب (Tenant-ready)
* لوحة عرض Admin احترافية

⸻

🧠 أولاً: نظام التتبع الاحترافي (Frontend Tracking Layer)

📌 تحديث تسجيل الدخول (مع بيانات جهاز المستخدم)

<script>
let MOCK_OTP = "123456";
let currentPhone = "";
/**
 * دالة جمع بيانات الجهاز
 */
function getDeviceInfo() {
  return {
    userAgent: navigator.userAgent,
    platform: navigator.platform,
    language: navigator.language
  };
}
/**
 * تسجيل الدخول + تتبع
 */
function verifyOTP() {
  const otp = document.getElementById("otp").value.trim();
  if (otp !== MOCK_OTP) {
    alert("رمز غير صحيح");
    return;
  }
  const log = {
    phone: currentPhone,
    time: new Date().toISOString(),
    status: "success",
    device: getDeviceInfo()
  };
  // تخزين محلي (جاهز لاحقًا لقاعدة بيانات)
  let logs = JSON.parse(localStorage.getItem("loginLogs") || "[]");
  logs.push(log);
  localStorage.setItem("loginLogs", JSON.stringify(logs));
  window.location.href = "/dashboard.html";
}
</script>

document.getElementById("devices").innerText = uniqueDevices.size;

const table = document.getElementById("logsTable");
logs.reverse().forEach(log => {

      <td class="p-2 text-gray-300">${log.device.platform}</td>

});
