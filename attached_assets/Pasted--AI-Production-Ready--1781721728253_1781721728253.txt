⚙️ ثالثاً: المحرك المالي (Financial Core Engine)

// FinancialCore.ts — Single Source of Truth
export type Money = number;
export interface TaxConfig {
  enabled: boolean;
  rate: number; // default 15%
}
export class FinancialCore {
  static calculateSubtotal(items: { quantity: number; unitPrice: number }[]): Money {
    return items.reduce((sum, i) => sum + i.quantity * i.unitPrice, 0);
  }
  static calculateTax(subtotal: Money, tax: TaxConfig): Money {
    if (!tax.enabled) return 0;
    return +(subtotal * (tax.rate / 100)).toFixed(2);
  }
  static calculateTotal(subtotal: Money, taxAmount: Money): Money {
    return +(subtotal + taxAmount).toFixed(2);
  }
  static format(amount: Money): string {
    return new Intl.NumberFormat('ar-SA', {
      style: 'currency',
      currency: 'SAR'
    }).format(amount);
  }
  static buildInvoice(items: any[], tax: TaxConfig) {
    const subtotal = this.calculateSubtotal(items);
    const taxAmount = this.calculateTax(subtotal, tax);
    const total = this.calculateTotal(subtotal, taxAmount);
    return {
      subtotal,
      taxAmount,
      total,
      currency: "SAR"
    };
  }
}

🌐 رابعاً: API النظام المالي (REST / Node.js)

// POST /api/invoices/create
export async function createInvoice(req, res) {
  const { items, tax_enabled, tax_rate, client_id } = req.body;
  const subtotal = FinancialCore.calculateSubtotal(items);
  const taxAmount = FinancialCore.calculateTax(subtotal, {
    enabled: tax_enabled,
    rate: tax_rate ?? 15
  });
  const total = FinancialCore.calculateTotal(subtotal, taxAmount);
  const invoice = await db.invoice.create({
    data: {
      client_id,
      subtotal,
      tax_enabled,
      tax_rate,
      tax_amount: taxAmount,
      total,
      currency: "SAR",
      status: "Draft"
    }
  });
  return res.json(invoice);
}
