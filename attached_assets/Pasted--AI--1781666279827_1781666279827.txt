🧠 2. Analyzer (Code Health Engine)

const fs = require("fs");
function analyzeProject(dir) {
  let files = 0;
  let lines = 0;
  let largeFiles = 0;
  function walk(d) {
    fs.readdirSync(d).forEach(f => {
      const full = `${d}/${f}`;
      if (fs.statSync(full).isDirectory()) return walk(full);
      if (f.endsWith(".js")) {
        files++;
        const content = fs.readFileSync(full, "utf8");
        const fileLines = content.split("\n").length;
        lines += fileLines;
        if (fileLines > 300) largeFiles++;
      }
    });
  }
  walk(dir);
  const score = Math.max(0, 100 - (largeFiles * 10));
  return {
    files,
    lines,
    largeFiles,
    codeHealthScore: score,
    techDebt: largeFiles > 5 ? "HIGH" : "MEDIUM"
  };
}
module.exports = { analyzeProject };

🔁 3. Duplicate Detector

const fs = require("fs");
const crypto = require("crypto");
function detectDuplicates(dir) {
  const map = {};
  function scan(d) {
    fs.readdirSync(d).forEach(f => {
      const full = `${d}/${f}`;
      if (fs.statSync(full).isDirectory()) return scan(full);
      const code = fs.readFileSync(full, "utf8");
      const hash = crypto.createHash("md5").update(code).digest("hex");
      if (!map[hash]) map[hash] = [];
      map[hash].push(full);
    });
  }
  scan(dir);
  const duplicates = Object.values(map).filter(g => g.length > 1);
  return {
    duplicateGroups: duplicates.length,
    files: duplicates
  };
}
module.exports = { detectDuplicates };

🧹 4. Auto Cleaner (آمن)

const fs = require("fs");
function cleanProject(dir) {
  let cleanedFiles = 0;
  function scan(d) {
    fs.readdirSync(d).forEach(f => {
      const full = `${d}/${f}`;
      if (fs.statSync(full).isDirectory()) return scan(full);
      if (!f.endsWith(".js")) return;
      let code = fs.readFileSync(full, "utf8");
      // تنظيف بسيط وآمن:
      code = code
        .replace(/console\.log\((.*?)\)/g, "// removed console.log")
        .replace(/\/\*[\s\S]*?\*\//g, ""); // remove comments
      fs.writeFileSync(full, code);
      cleanedFiles++;
    });
  }
  scan(dir);
  return {
    cleanedFiles
  };
}
module.exports = { cleanProject };
