🚨 2. ANOMALY DETECTION ENGINE

يرصد المشاكل قبل ما تتحول إلى انهيار

function detectAnomaly(metrics) {
  return {
    stripeFailureSpike: metrics.stripeErrors > 5,
    dbLatencyHigh: metrics.dbLatency > 800,
    apiErrorGrowth: metrics.errorRate > 0.03,
    tenantLeakRisk: metrics.missingTenantChecks > 0
  };
}

⸻

🧯 3. SELF-HEALING CONTROLLER (القلب الحقيقي)

لا يصلح كل شيء—فقط “المسموح إصلاحه”

export async function selfHeal(anomaly) {
  if (anomaly.stripeFailureSpike) {
    return handleStripeRecovery();
  }
  if (anomaly.dbLatencyHigh) {
    return reduceQueryLoad();
  }
  if (anomaly.apiErrorGrowth) {
    return rollbackLastSafeState();
  }
  if (anomaly.tenantLeakRisk) {
    return enforceIsolationMode();
  }
}
