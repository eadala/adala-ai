🧭 2️⃣ Routing الرئيسي (App Router)

import { BrowserRouter, Routes, Route } from "react-router-dom";
import OfficeRoutes from "./app/routes/OfficeRoutes";
import OwnerRoutes from "./app/routes/OwnerRoutes";
import DevRoutes from "./app/routes/DevRoutes";
export default function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/office/*" element={<OfficeRoutes />} />
        <Route path="/owner/*" element={<OwnerRoutes />} />
        <Route path="/dev/*" element={<DevRoutes />} />
      </Routes>
    </BrowserRouter>
  );
}

🔐 6️⃣ Role-Based Route Guard (مهم جداً)

import { Navigate } from "react-router-dom";
export default function RoleRoute({ role, allowed, children }) {
  if (!allowed.includes(role)) {
    return <Navigate to="/unauthorized" />;
  }
  return children;
}

⸻

مثال استخدام

<RoleRoute role={user.role} allowed={["firm_owner"]}>
  <OwnerDashboard />
</RoleRoute>

⸻

🧠 7️⃣ Permission Gate (داخل الصفحات)

export function PermissionGate({ permission, children }) {
  const { permissions } = usePermissions();
  if (!permissions.includes(permission)) return null;
  return children;
}

⸻

مثال

<PermissionGate permission="cases:delete">
  <DeleteCaseButton />
</PermissionGate>
