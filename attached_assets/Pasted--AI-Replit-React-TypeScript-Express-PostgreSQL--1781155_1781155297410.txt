# 1. جدول الملفات

## shared/schema/caseDocuments.ts

```ts
import {
  pgTable,
  uuid,
  varchar,
  bigint,
  boolean,
  timestamp
} from "drizzle-orm/pg-core";

export const caseDocuments = pgTable(
  "case_documents",
  {

    id: uuid("id")
      .defaultRandom()
      .primaryKey(),

    officeId: uuid("office_id")
      .notNull(),

    caseId: uuid("case_id")
      .notNull(),

    fileName: varchar(
      "file_name",
      {
        length: 500
      }
    ),

    originalName: varchar(
      "original_name",
      {
        length: 500
      }
    ),

    mimeType: varchar(
      "mime_type",
      {
        length: 100
      }
    ),

    fileSize: bigint(
      "file_size",
      {
        mode: "number"
      }
    ),

    hash: varchar(
      "hash",
      {
        length: 255
      }
    ),

    fileUrl: varchar(
      "file_url",
      {
        length: 1000
      }
    ),

    storageProvider: varchar(
      "storage_provider",
      {
        length: 100
      }
    ).default("cloudflare-r2"),

    archived: boolean(
      "archived"
    ).default(false),

    deleted: boolean(
      "deleted"
    ).default(false),

    createdAt: timestamp(
      "created_at"
    ).defaultNow()

  }
);
```

# 2. Deduplication

## server/services/hashService.ts

```ts
import crypto from "crypto";
import fs from "fs";

export function generateHash(
filePath:string
){

const buffer =
fs.readFileSync(filePath);

return crypto
.createHash("sha256")
.update(buffer)
.digest("hex");

}
```

# 3. منع تكرار الملفات

## storageService.ts

```ts
import { db } from "../db";
import { caseDocuments } from "@shared/schema";
import { eq } from "drizzle-orm";

export async function checkDuplicate(
hash:string
){

const file =
await db.query.caseDocuments.findFirst({

where:eq(
caseDocuments.hash,
hash
)

});

return file;

}
```

# 5. الأرشفة الذكية

## archiveService.ts

```ts
import { db } from "../db";
import { caseDocuments } from "@shared/schema";
import { eq } from "drizzle-orm";

export async function archiveFile(
id:string
){

await db.update(
caseDocuments
)

.set({

archived:true

})

.where(
eq(
caseDocuments.id,
id
)
);

}
```

# 7. النسخ الاحتياطي إلى Telegram

## telegramBackupService.ts

```ts
import axios from "axios";
import fs from "fs";
import FormData from "form-data";

export async function uploadBackup(

filePath:string,

chatId:string

){

const form =
new FormData();

form.append(
"chat_id",
chatId
);

form.append(
"document",
fs.createReadStream(
filePath
)
);

await axios.post(

`https://api.telegram.org/bot${process.env.TELEGRAM_TOKEN}/sendDocument`,

form,

{
headers:
form.getHeaders()
}

);

}
```
