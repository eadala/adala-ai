🧱 2) قاعدة البيانات (Multi-Tenant Schema)

model Tenant {
  id        String   @id @default(cuid())
  name      String
  domain    String?  // custom domain
  plan      String   @default("FREE")
  stripeId  String?
  createdAt DateTime @default(now())
  users     User[]
  cases     Case[]
  invoices  Invoice[]
}
model User {
  id        String  @id @default(cuid())
  tenantId  String
  email     String
  role      String
  plan      String
  tenant    Tenant  @relation(fields: [tenantId], references: [id])
}
model Case {
  id        String  @id @default(cuid())
  tenantId  String
  title     String
  status    String
  tenant    Tenant  @relation(fields: [tenantId], references: [id])
}
model Invoice {
  id        String  @id @default(cuid())
  tenantId  String
  amount    Float
  status    String
  tenant    Tenant  @relation(fields: [tenantId], references: [id])
}

🔐 3) Middleware العزل (Tenant Isolation)

import { Request, Response, NextFunction } from "express";
/**
 * استخراج الـ Tenant من الطلب
 */
export function tenantMiddleware(req: any, res: Response, next: NextFunction) {
  const tenantId =
    req.headers["x-tenant-id"] ||
    req.user?.tenantId;
  if (!tenantId) {
    return res.status(401).json({ error: "Tenant not found" });
  }
  req.tenantId = tenantId;
  next();
}

🧠 4) Feature Flags لكل Tenant

export const TENANT_PLANS = {
  FREE: {
    features: ["WEBSITE", "STORE", "PAYMENTS_BASIC"],
    limits: { users: 2, ai: 5 }
  },
  PRO: {
    features: ["WEBSITE", "STORE", "PAYMENTS", "AI_ANALYTICS"],
    limits: { users: 10, ai: 100 }
  },
  ENTERPRISE: {
    features: ["ALL"],
    limits: { users: -1, ai: -1 }
  }
};
export function getTenantPlan(tenant: any) {
  return TENANT_PLANS[tenant.plan];
}
export function hasFeature(tenant: any, feature: string) {
  const plan = getTenantPlan(tenant);
  if (plan.features.includes("ALL")) return true;
  return plan.features.includes(feature);
}

💳 5) Stripe لكل Tenant (Subscription Binding)

import stripe from "./stripe";
/**
 * إنشاء اشتراك لكل مكتب (Tenant)
 */
export async function createTenantSubscription(tenantId: string, priceId: string) {
  const session = await stripe.checkout.sessions.create({
    mode: "subscription",
    payment_method_types: ["card"],
    line_items: [{ price: priceId, quantity: 1 }],
    success_url: `${process.env.APP_URL}/success`,
    cancel_url: `${process.env.APP_URL}/cancel`,
    metadata: {
      tenantId
    }
  });
  return session.url;
}

🔁 6) Webhook Multi-Tenant (أهم جزء)

import stripe from "./stripe";
export async function handleWebhook(req: any, res: any) {
  const event = stripe.webhooks.constructEvent(
    req.body,
    req.headers["stripe-signature"],
    process.env.STRIPE_WEBHOOK_SECRET!
  );
  switch (event.type) {
    // 💳 تفعيل الاشتراك
    case "checkout.session.completed": {
      const session = event.data.object;
      const tenantId = session.metadata.tenantId;
      await db.tenant.update({
        where: { id: tenantId },
        data: {
          plan: "PRO",
          stripeId: session.customer
        }
      });
      break;
    }
    // ❌ فشل الدفع
    case "invoice.payment_failed": {
      const invoice = event.data.object;
      const tenantId = invoice.metadata?.tenantId;
      if (tenantId) {
        await db.tenant.update({
          where: { id: tenantId },
          data: { plan: "FREE" } // downgrade تلقائي
        });
      }
      break;
    }
  }
  res.json({ received: true });
}

⚙️ 7) Repository Layer (عزل البيانات لكل مكتب)

export class TenantRepository {
  static async getCases(tenantId: string) {
    return db.case.findMany({
      where: { tenantId }
    });
  }
  static async createCase(tenantId: string, data: any) {
    return db.case.create({
      data: {
        ...data,
        tenantId
      }
    });
  }
  static async getInvoices(tenantId: string) {
    return db.invoice.findMany({
      where: { tenantId }
    });
  }
}

💰 9) Revenue Isolation (لكل Tenant)

export async function calculateTenantRevenue(tenantId: string) {
  const invoices = await db.invoice.findMany({
    where: { tenantId }
  });
  const total = invoices.reduce((sum, i) => sum + i.amount, 0);
  const platformFee = total * 0.10;
  const stripeFee = total * 0.029;
  return {
    gross: total,
    platformFee,
    stripeFee,
    net: total - platformFee - stripeFee
  };
}

  const cases = await TenantRepository.getCases(req.tenantId);
  res.json(cases);
});
/**
 * إنشاء قضية
 */
app.post("/cases", async (req: any, res) => {
  const result = await TenantRepository.createCase(req.tenantId, req.body);
  res.json(result);
});
app.listen(3000, () => {
  console.log("🚀 Multi-Tenant SaaS Running");
});
