3) ⚙️ محرك الفواتير المالي (TypeScript Core Engine)

// /lib/financial/FinancialCore.ts
export type Money = number;
export interface TaxConfig {
  enabled: boolean;
  rate: number; // e.g. 15
}
export class FinancialCore {
  static calculateSubtotal(items: {
    quantity: number;
    unitPrice: number;
  }[]): Money {
    return items.reduce((sum, item) => {
      return sum + item.quantity * item.unitPrice;
    }, 0);
  }
  static calculateTax(subtotal: Money, tax: TaxConfig): Money {
    if (!tax.enabled) return 0;
    return +(subtotal * (tax.rate / 100)).toFixed(2);
  }
  static calculateTotal(subtotal: Money, taxAmount: Money): Money {
    return +(subtotal + taxAmount).toFixed(2);
  }
  static formatSAR(amount: Money): string {
    return new Intl.NumberFormat('ar-SA', {
      style: 'currency',
      currency: 'SAR'
    }).format(amount);
  }
  static buildInvoice(items: any[], tax: TaxConfig) {
    const subtotal = this.calculateSubtotal(items);
    const taxAmount = this.calculateTax(subtotal, tax);
    const total = this.calculateTotal(subtotal, taxAmount);
    return {
      subtotal,
      taxAmount,
      total,
      currency: "SAR"
    };
  }
}
