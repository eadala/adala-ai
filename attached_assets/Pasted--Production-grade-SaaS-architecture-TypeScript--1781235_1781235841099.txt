🔐 4) Feature Guard (التحكم الحقيقي)

import { PLANS } from "./plans";
import { Feature, User } from "./types";
export class FeatureGuard {
  static hasFeature(user: User, feature: Feature): boolean {
    const plan = PLANS[user.plan];
    return plan.features.includes(feature);
  }
  static checkLimit(
    user: User,
    key: keyof typeof PLANS["FREE"]["limits"],
    currentValue: number
  ): boolean {
    const plan = PLANS[user.plan];
    const limit = plan.limits[key];
    if (limit === -1) return true;
    return currentValue < limit;
  }
  static assertFeature(user: User, feature: Feature) {
    if (!this.hasFeature(user, feature)) {
      throw new Error(`Feature ${feature} not available in ${user.plan}`);
    }
  }
  static assertLimit(
    user: User,
    key: keyof typeof PLANS["FREE"]["limits"],
    currentValue: number
  ) {
    if (!this.checkLimit(user, key, currentValue)) {
      throw new Error(`Limit exceeded for ${key}`);
    }
  }
}

🤖 5) Usage Metering (AI + النظام)

import { User } from "./types";
export class UsageMeter {
  private static todayKey(userId: string) {
    const d = new Date().toISOString().split("T")[0];
    return `${userId}:${d}`;
  }
  static incrementAI(user: User, userId: string): number {
    const key = this.todayKey(userId);
    user.usage[key] = (user.usage[key] || 0) + 1;
    return user.usage[key];
  }
  static getAIUsage(user: User, userId: string): number {
    const key = this.todayKey(userId);
    return user.usage[key] || 0;
  }
}

🧠 6) AI Gate (تحكم ذكي كامل)

import { FeatureGuard } from "./guard";
import { UsageMeter } from "./metering";
import { FEATURES, User } from "./types";
export function runAI(user: User, userId: string, input: string) {
  // 1. تحقق من الميزة
  FeatureGuard.assertFeature(user, FEATURES.AI_ANALYTICS);
  // 2. تحقق من الحد
  const usage = UsageMeter.getAIUsage(user, userId);
  FeatureGuard.assertLimit(user, "aiCallsPerDay", usage);
  // 3. تسجيل الاستخدام
  UsageMeter.incrementAI(user, userId);
  // 4. تنفيذ AI
  return {
    result: `AI processed: ${input}`,
    usage: usage + 1
  };
}
