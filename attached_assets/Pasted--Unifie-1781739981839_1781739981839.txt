⚙️ 2) Case Engine (القلب التشغيلي)

export async function getCaseFull(caseId) {
  const caseData = await db.query(`SELECT * FROM cases WHERE id=$1`, [caseId]);
  const hearings = await db.query(`SELECT * FROM hearings WHERE case_id=$1`, [caseId]);
  return {
    ...caseData.rows[0],
    hearings: hearings.rows,
  };
}

⸻

📅 3) الجلسة القادمة (Auto System)

export function getNextHearing(hearings) {
  return hearings
    .filter(h => new Date(h.hearing_date) > new Date())
    .sort((a,b) => new Date(a.hearing_date) - new Date(b.hearing_date))[0];
}

⸻

export async function syncCase(caseId) {
  const caseData = await getCaseFull(caseId);
  const next = getNextHearing(caseData.hearings);
  await db.query(
    `UPDATE cases SET next_hearing_date=$1 WHERE id=$2`,
    [next?.hearing_date || null, caseId]
  );
}

🔔 5) Notification Engine

function getAlertLevel(date) {
  const diff = new Date(date) - new Date();
  if (diff < 7 * 86400000) return "7_days";
  if (diff < 3 * 86400000) return "3_days";
  if (diff < 86400000) return "24_hours";
}
