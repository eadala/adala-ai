🧩 المرحلة 4: بناء أول Module (Cases)

src/modules/cases/controller.ts

import { prisma } from "../../db/prisma";
export const createCase = async (req, res) => {
  const { title, tenantId } = req.body;
  const newCase = await prisma.case.create({
    data: {
      title,
      status: "OPEN",
      tenantId,
    },
  });
  res.json(newCase);
};

⸻

src/api/cases.ts

import express from "express";
import { createCase } from "../modules/cases/controller";
const router = express.Router();
router.post("/", createCase);
export default router;

⸻

ربطه في app.ts

import casesRouter from "./api/cases";
app.use("/api/cases", casesRouter);

🤖 المرحلة 6: إدخال AI (بسيط أولاً)

src/ai/generator.ts

export const generateLegalDoc = async (prompt: string) => {
  const response = await fetch("https://generativelanguage.googleapis.com/...", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
    },
    body: JSON.stringify({ prompt }),
  });
  return response.json();
};

});

🔐 المرحلة 7: Multi-Tenant Middleware

src/middleware/tenant.ts

export const tenantMiddleware = (req, res, next) => {
  const tenantId = req.headers["x-tenant-id"];
  if (!tenantId) {
    return res.status(400).json({ error: "Missing tenant" });
  }
  req.tenantId = tenantId;
  next();
};

⸻

استخدامه

app.use(tenantMiddleware);
