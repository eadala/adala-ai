## 1. جدول الباقات

### shared/schema/plans.ts

```ts
import {
  pgTable,
  uuid,
  varchar,
  decimal,
  boolean,
  integer
} from "drizzle-orm/pg-core";

export const plans = pgTable("plans", {

  id: uuid("id")
    .defaultRandom()
    .primaryKey(),

  name: varchar("name", {
    length: 100
  }).notNull(),

  slug: varchar("slug", {
    length: 100
  }).notNull(),

  monthlyPrice: decimal(
    "monthly_price"
  ),

  yearlyPrice: decimal(
    "yearly_price"
  ),

  displayOrder: integer(
    "display_order"
  ).default(1),

  recommended: boolean(
    "recommended"
  ).default(false),

  isVisible: boolean(
    "is_visible"
  ).default(true)

});
```

## إضافة باقة

```ts
router.post("/", async (req, res) => {

  const {

    name,

    slug,

    monthlyPrice,

    yearlyPrice

  } = req.body;

  await db.insert(plans).values({

    name,

    slug,

    monthlyPrice,

    yearlyPrice

  });

  res.json({
    success: true
  });

});
```

---

# تعديل الباقة

```ts
router.patch("/:id", async (req, res) => {

  await db
    .update(plans)
    .set(req.body)
    .where(
      eq(
        plans.id,
        req.params.id
      )
    );

  res.json({
    success: true
  });

});
```

---

# حذف الباقة

```ts
router.delete("/:id", async (req, res) => {

  await db
    .delete(plans)
    .where(
      eq(
        plans.id,
        req.params.id
      )
    );

  res.json({
    success: true
  });

});
```

# صلاحية خاصة

```ts
SUPER_ADMIN
```

ويستطيع:

* إنشاء عدد لا محدود من الباقات.
* تعديل الأسعار.
* إضافة مزايا جديدة.
* حذف الباقات.
* إخفاء أو إظهار الباقات.
* تحديد الباقة الموصى بها.
* منح اشتراكات مجانية.
* ترقية أو تخفيض أي مكتب.
* إنشاء كوبونات خصم.
* إدارة الإضافات Add-ons.

وهذا التصميم قريب جداً من لوحة إدارة الاشتراكات الموجودة في Stripe Billing وHubSpot وShopify، ومناسب لبناء عدالة AI كنظام SaaS احترافي قابل للتوسع.
