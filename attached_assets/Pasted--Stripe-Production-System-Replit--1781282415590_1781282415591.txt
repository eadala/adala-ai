💳 4) إنشاء Checkout Session (اشتراك)

const stripe = require("./stripe");
const { PLANS } = require("./plans");
async function createCheckoutSession(userId, planKey) {
  const plan = PLANS[planKey];
  if (!plan) throw new Error("Invalid plan");
  const session = await stripe.checkout.sessions.create({
    mode: "subscription",
    payment_method_types: ["card"],
    line_items: [
      {
        price: plan.priceId,
        quantity: 1
      }
    ],
    success_url: "http://localhost:3000/success",
    cancel_url: "http://localhost:3000/cancel",
    metadata: {
      userId,
      planKey
    }
  });
  return session.url;
}
module.exports = { createCheckoutSession };

🌐 5) Webhook (الأهم في النظام)

const stripe = require("./stripe");
const { PLANS, PLATFORM_FEE } = require("./plans");
/**
 * قاعدة بيانات بسيطة (استبدلها بـ PostgreSQL لاحقًا)
 */
const DB = {
  users: {}
};
function saveUserPlan(userId, plan) {
  DB.users[userId] = {
    plan,
    updatedAt: new Date()
  };
}
function calculateRevenue(amount) {
  const platformFee = amount * PLATFORM_FEE;
  const stripeFee = amount * 0.029;
  const net = amount - platformFee - stripeFee;
  return { platformFee, stripeFee, net };
}
async function handleWebhook(req, res) {
  const sig = req.headers["stripe-signature"];
  let event;
  try {
    event = stripe.webhooks.constructEvent(
      req.body,
      sig,
      process.env.STRIPE_WEBHOOK_SECRET
    );
  } catch (err) {
    return res.status(400).send(`Webhook Error: ${err.message}`);
  }
  switch (event.type) {
    // 💳 نجاح الاشتراك
    case "checkout.session.completed": {
      const session = event.data.object;
      const userId = session.metadata.userId;
      const plan = session.metadata.planKey;
      saveUserPlan(userId, plan);
      break;
    }
    // 💰 فاتورة مدفوعة
    case "invoice.paid": {
      const invoice = event.data.object;
      const amount = invoice.amount_paid / 100;
      const revenue = calculateRevenue(amount);
      console.log("💰 Revenue Report:", revenue);
      break;
    }
    // ❌ فشل الدفع
    case "invoice.payment_failed": {
      console.log("Payment failed");
      break;
    }
  }
  res.json({ received: true });
}
module.exports = { handleWebhook, DB };

🚀 6) السيرفر الرئيسي (Express App)

require("dotenv").config();
const express = require("express");
const stripe = require("./stripe");
const { createCheckoutSession } = require("./checkout");
const { handleWebhook } = require("./webhook");
const app = express();
// مهم للـ Stripe webhook
app.post("/webhook", express.raw({ type: "application/json" }), handleWebhook);
app.use(express.json());
/**
 * 💳 إنشاء اشتراك
 */
app.post("/api/subscribe", async (req, res) => {
  const { userId, plan } = req.body;
  try {
    const url = await createCheckoutSession(userId, plan);
    res.json({ checkoutUrl: url });
  } catch (err) {
    res.status(400).json({ error: err.message });
  }
});
/**
 * 📊 حساب العمولة قبل الدفع (عرض شفافية)
 */
app.post("/api/calc", (req, res) => {
  const amount = Number(req.body.amount);
  const platformFee = amount * 0.10;
  const stripeFee = amount * 0.029;
  const net = amount - platformFee - stripeFee;
  res.json({
    amount,
    platformFee,
    stripeFee,
    net
  });
});
/**
 * 👤 جلب الباقة الحالية
 */
app.get("/api/user/:id", (req, res) => {
  const { DB } = require("./webhook");
  const user = DB.users[req.params.id];
  res.json(user || { plan: "FREE" });
});
app.listen(3000, () => {
  console.log("🚀 Server running on http://localhost:3000");
});
